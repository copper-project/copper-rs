68/// It will create a new type called CuStampedDataSet you can pass to the log reader for decoding:
69#[proc_macro]
70pub fn gen_cumsgs(config_path_lit: TokenStream) -> TokenStream {
71    #[cfg(feature = "std")]
72    let std = true;
73
74    #[cfg(not(feature = "std"))]
75    let std = false;
76
77    let config = parse_macro_input!(config_path_lit as LitStr).value();
78    if !std::path::Path::new(&config_full_path(&config)).exists() {
79        return return_error(format!(
80            "The configuration file `{config}` does not exist. Please provide a valid path."
81        ));
82    }
83    #[cfg(feature = "macro_debug")]
84    eprintln!("[gen culist support with {config:?}]");
85    let cuconfig = match read_config(&config) {
86        Ok(cuconfig) => cuconfig,
87        Err(e) => return return_error(e.to_string()),
88    };
89    let graph = cuconfig
90        .get_graph(None) // FIXME(gbin): Multimission
91        .expect("Could not find the specified mission for gen_cumsgs");
92    let task_specs = CuTaskSpecSet::from_graph(graph);

94    let mut bridge_specs = build_bridge_specs(&cuconfig, graph, &channel_usage);

99        };
100    let task_member_names = collect_task_member_names(graph);
101    let (culist_order, node_output_positions) = collect_culist_metadata(
102        &culist_plan,

139            use cu29::bincode::enc::Encoder;
140            use cu29::bincode::error::EncodeError;

162/// Build the inner support of the copper list.
163fn gen_culist_support(
164    runtime_plan: &CuExecutionLoop,
165    culist_indices_in_plan_order: &[usize],
166    node_output_positions: &HashMap<NodeId, usize>,
167    task_member_names: &[(NodeId, String)],
168) -> proc_macro2::TokenStream {
169    #[cfg(feature = "macro_debug")]
170    eprintln!("[Extract msgs types]");
171    let output_packs = extract_output_packs(runtime_plan);
172    let slot_types: Vec<Type> = output_packs.iter().map(|pack| pack.slot_type()).collect();
173
174    let culist_size = output_packs.len();
175
176    #[cfg(feature = "macro_debug")]
177    eprintln!("[build the copperlist struct]");
178    let msgs_types_tuple: TypeTuple = build_culist_tuple(&slot_types);
179
180    #[cfg(feature = "macro_debug")]
181    eprintln!("[build the copperlist tuple bincode support]");
182    let msgs_types_tuple_encode = build_culist_tuple_encode(&slot_types);
183    let msgs_types_tuple_decode = build_culist_tuple_decode(&slot_types);
184
185    #[cfg(feature = "macro_debug")]
186    eprintln!("[build the copperlist tuple debug support]");
187    let msgs_types_tuple_debug = build_culist_tuple_debug(&slot_types);
188
189    #[cfg(feature = "macro_debug")]
190    eprintln!("[build the copperlist tuple serialize support]");
191    let msgs_types_tuple_serialize = build_culist_tuple_serialize(&slot_types);
192
193    #[cfg(feature = "macro_debug")]
194    eprintln!("[build the default tuple support]");
195    let msgs_types_tuple_default = build_culist_tuple_default(&slot_types);
196
197    #[cfg(feature = "macro_debug")]
198    eprintln!("[build erasedcumsgs]");
199
200    let erasedmsg_trait_impl = build_culist_erasedcumsgs(&output_packs);
201
202    let metadata_accessors: Vec<proc_macro2::TokenStream> = culist_indices_in_plan_order
203        .iter()
204        .map(|idx| {
205            let slot_index = syn::Index::from(*idx);
206            let pack = output_packs
207                .get(*idx)
208                .unwrap_or_else(|| panic!("Missing output pack for index {idx}"));
209            if pack.is_multi() {
210                quote! { &culist.msgs.0.#slot_index.0.metadata }
211            } else {
212                quote! { &culist.msgs.0.#slot_index.metadata }
213            }
214        })
215        .collect();
216    let mut zeroed_init_tokens: Vec<proc_macro2::TokenStream> = Vec::new();
217    for idx in culist_indices_in_plan_order {
218        let slot_index = syn::Index::from(*idx);
219        let pack = output_packs
220            .get(*idx)
221            .unwrap_or_else(|| panic!("Missing output pack for index {idx}"));
222        if pack.is_multi() {
223            for port_idx in 0..pack.msg_types.len() {
224                let port_index = syn::Index::from(port_idx);
225                zeroed_init_tokens.push(quote! {
226                    self.0.#slot_index.#port_index.metadata.status_txt = CuCompactString::default();
227                });
228            }
229        } else {
230            zeroed_init_tokens.push(quote! {
231                self.0.#slot_index.metadata.status_txt = CuCompactString::default();
232            });
233        }
234    }
235    let collect_metadata_function = quote! {
236        pub fn collect_metadata<'a>(culist: &'a CuList) -> [&'a CuMsgMetadata; #culist_size] {
237            [#( #metadata_accessors, )*]
238        }
239    };
240
241    let payload_bytes_accumulators: Vec<proc_macro2::TokenStream> = culist_indices_in_plan_order
242        .iter()
243        .map(|idx| {
244            let slot_index = syn::Index::from(*idx);
245            let pack = output_packs
246                .get(*idx)
247                .unwrap_or_else(|| panic!("Missing output pack for index {idx}"));
248            if pack.is_multi() {
249                let iter = (0..pack.msg_types.len()).map(|port_idx| {
250                    let port_index = syn::Index::from(port_idx);
251                    quote! {
252                        if let Some(payload) = culist.msgs.0.#slot_index.#port_index.payload() {
253                            raw += cu29::monitoring::CuPayloadSize::raw_bytes(payload);
254                            handles += cu29::monitoring::CuPayloadSize::handle_bytes(payload);
255                        }
256                    }
257                });
258                quote! { #(#iter)* }
259            } else {
260                quote! {
261                    if let Some(payload) = culist.msgs.0.#slot_index.payload() {
262                        raw += cu29::monitoring::CuPayloadSize::raw_bytes(payload);
263                        handles += cu29::monitoring::CuPayloadSize::handle_bytes(payload);
264                    }
265                }
266            }
267        })
268        .collect();
269
270    let compute_payload_bytes_fn = quote! {
271        pub fn compute_payload_bytes(culist: &CuList) -> (u64, u64) {
272            let mut raw: usize = 0;
273            let mut handles: usize = 0;
274            #(#payload_bytes_accumulators)*
275            (raw as u64, handles as u64)
276        }
277    };
278
279    let task_name_literals: Vec<String> = task_member_names
280        .iter()
281        .map(|(_, name)| name.clone())
282        .collect();
283
284    let mut methods = Vec::new();
285    for (node_id, name) in task_member_names {
286        let output_position = node_output_positions
287            .get(node_id)
288            .unwrap_or_else(|| panic!("Task {name} (id: {node_id}) not found in execution order"));
289        let pack = output_packs
290            .get(*output_position)
291            .unwrap_or_else(|| panic!("Missing output pack for task {name}"));
292        let slot_index = syn::Index::from(*output_position);
293
294        if pack.msg_types.len() == 1 {
295            let fn_name = format_ident!("get_{}_output", name);
296            let payload_type = pack.msg_types.first().unwrap();
297            methods.push(quote! {
298                #[allow(dead_code)]
299                pub fn #fn_name(&self) -> &CuMsg<#payload_type> {
300                    &self.0.#slot_index
301                }
302            });
303        } else {
304            let outputs_fn = format_ident!("get_{}_outputs", name);
305            let slot_type = pack.slot_type();
306            for (port_idx, payload_type) in pack.msg_types.iter().enumerate() {
307                let fn_name = format_ident!("get_{}_output_{}", name, port_idx);
308                let port_index = syn::Index::from(port_idx);
309                methods.push(quote! {
310                    #[allow(dead_code)]
311                    pub fn #fn_name(&self) -> &CuMsg<#payload_type> {
312                        &self.0.#slot_index.#port_index
313                    }
314                });
315            }
316            methods.push(quote! {
317                #[allow(dead_code)]
318                pub fn #outputs_fn(&self) -> &#slot_type {
319                    &self.0.#slot_index
320                }
321            });
322        }
323    }
324
325    // This generates a way to get the metadata of every single message of a culist at low cost
326    quote! {
327        #collect_metadata_function
328        #compute_payload_bytes_fn
329
330        pub struct CuStampedDataSet(pub #msgs_types_tuple);
331
332        pub type CuList = CopperList<CuStampedDataSet>;
333
334        impl CuStampedDataSet {
335            #(#methods)*
336
337            #[allow(dead_code)]
338            fn get_tuple(&self) -> &#msgs_types_tuple {
339                &self.0
340            }
341
342            #[allow(dead_code)]
343            fn get_tuple_mut(&mut self) -> &mut #msgs_types_tuple {
344                &mut self.0
345            }
346        }
347
348        impl MatchingTasks for CuStampedDataSet {
349            #[allow(dead_code)]
350            fn get_all_task_ids() -> &'static [&'static str] {
351                &[#(#task_name_literals),*]
352            }
353        }
354
355        // Note: PayloadSchemas is NOT implemented here.
356        // Users who want MCAP export with schemas should implement it manually
357        // using cu29_export::serde_to_jsonschema::trace_type_to_jsonschema.
358
359        // Adds the bincode support for the copper list tuple
360        #msgs_types_tuple_encode
361        #msgs_types_tuple_decode
362
363        // Adds the debug support
364        #msgs_types_tuple_debug
365
366        // Adds the serialization support
367        #msgs_types_tuple_serialize
368
369        // Adds the default support
370        #msgs_types_tuple_default
371
372        // Adds the type erased CuStampedDataSet support (to help generic serialized conversions)
373        #erasedmsg_trait_impl
374
375        impl CuListZeroedInit for CuStampedDataSet {
376            fn init_zeroed(&mut self) {
377                #(#zeroed_init_tokens)*
378            }
379        }
380    }
381}

383fn gen_sim_support(
384    runtime_plan: &CuExecutionLoop,
385    exec_entities: &[ExecutionEntity],
386) -> proc_macro2::TokenStream {
387    #[cfg(feature = "macro_debug")]
388    eprintln!("[Sim: Build SimEnum]");
389    let plan_enum: Vec<proc_macro2::TokenStream> = runtime_plan
390        .steps
391        .iter()
392        .filter_map(|unit| match unit {
393            CuExecutionUnit::Step(step) => {
394                if !matches!(
395                    exec_entities[step.node_id as usize].kind,
396                    ExecutionEntityKind::Task { .. }
397                ) {
398                    return None;
399                }
400                let enum_entry_name = config_id_to_enum(step.node.get_id().as_str());
401                let enum_ident = Ident::new(&enum_entry_name, Span::call_site());
402                let inputs: Vec<Type> = step
403                    .input_msg_indices_types
404                    .iter()
405                    .map(|input| {
406                        parse_str::<Type>(format!("CuMsg<{}>", input.msg_type).as_str()).unwrap()
407                    })
408                    .collect();
409                let output: Option<Type> = step.output_msg_pack.as_ref().map(|pack| {
410                    let msg_types: Vec<Type> = pack
411                        .msg_types
412                        .iter()
413                        .map(|msg_type| {
414                            parse_str::<Type>(msg_type.as_str()).unwrap_or_else(|_| {
415                                panic!("Could not transform {msg_type} into a message Rust type.")
416                            })
417                        })
418                        .collect();
419                    build_output_slot_type(&msg_types)
420                });
421                let no_output = parse_str::<Type>("CuMsg<()>").unwrap();
422                let output = output.as_ref().unwrap_or(&no_output);
423
424                let inputs_type = if inputs.is_empty() {
425                    quote! { () }
426                } else if inputs.len() == 1 {
427                    let input = inputs.first().unwrap();
428                    quote! { &'a #input }
429                } else {
430                    quote! { &'a (#(&'a #inputs),*) }
431                };
432
433                Some(quote! {
434                    #enum_ident(CuTaskCallbackState<#inputs_type, &'a mut #output>)
435                })
436            }
437            CuExecutionUnit::Loop(_) => {
438                todo!("Needs to be implemented")
439            }
440        })
441        .collect();
442    let mut variants = plan_enum;
443    variants.push(quote! { __Phantom(core::marker::PhantomData<&'a ()>) });
444    quote! {
445        // not used if sim is not generated but this is ok.
446        #[allow(dead_code, unused_lifetimes)]
447        pub enum SimStep<'a> {
448            #(#variants),*
449        }
450    }
451}

453/// Adds #[copper_runtime(config = "path", sim_mode = false/true)] to your application struct to generate the runtime.
454/// if sim_mode is omitted, it is set to false.
455/// This will add a "runtime" field to your struct and implement the "new" and "run" methods.
456#[proc_macro_attribute]
457pub fn copper_runtime(args: TokenStream, input: TokenStream) -> TokenStream {
458    #[cfg(feature = "macro_debug")]
459    eprintln!("[entry]");
460    let mut application_struct = parse_macro_input!(input as ItemStruct);
461
462    let application_name = &application_struct.ident;
463    let builder_name = format_ident!("{}Builder", application_name);
464
465    let mut config_file: Option<LitStr> = None;
466    let mut sim_mode = false;
467
468    #[cfg(feature = "std")]
469    let std = true;
470
471    #[cfg(not(feature = "std"))]
472    let std = false;
473
474    let rt_guard = rtsan_guard_tokens();
475
476    // Custom parser for the attribute arguments
477    let attribute_config_parser = parser(|meta| {
478        if meta.path.is_ident("config") {
479            config_file = Some(meta.value()?.parse()?);
480            Ok(())
481        } else if meta.path.is_ident("sim_mode") {
482            // Check if `sim_mode` has an explicit value (true/false)
483            if meta.input.peek(syn::Token![=]) {
484                meta.input.parse::<syn::Token![=]>()?;
485                let value: syn::LitBool = meta.input.parse()?;
486                sim_mode = value.value();
487                Ok(())
488            } else {
489                // If no value is provided, default to true
490                sim_mode = true;
491                Ok(())
492            }
493        } else {
494            Err(meta.error("unsupported property"))
495        }
496    });
497
498    #[cfg(feature = "macro_debug")]
499    eprintln!("[parse]");
500    // Parse the provided args with the custom parser
501    parse_macro_input!(args with attribute_config_parser);
502
503    // Adds the generic parameter for the UnifiedLogger if this is a real application (not sim)
504    // This allows to adapt either to the no-std (custom impl) and std (default file based one)
505    // if !sim_mode {
506    //     application_struct
507    //         .generics
508    //         .params
509    //         .push(syn::parse_quote!(L: UnifiedLogWrite + 'static));
510    // }
511
512    // Check if the config file was provided
513    let config_file = match config_file {
514        Some(file) => file.value(),
515        None => {
516            return return_error(
517                "Expected config file attribute like #[CopperRuntime(config = \"path\")]"
518                    .to_string(),
519            );
520        }
521    };
522
523    if !std::path::Path::new(&config_full_path(&config_file)).exists() {
524        return return_error(format!(
525            "The configuration file `{config_file}` does not exist. Please provide a valid path."
526        ));
527    }
528
529    let copper_config = match read_config(&config_file) {
530        Ok(cuconfig) => cuconfig,
531        Err(e) => return return_error(e.to_string()),
532    };
533    let copper_config_content = match read_to_string(config_full_path(config_file.as_str())) {
534        Ok(ok) => ok,
535        Err(e) => {
536            return return_error(format!(
537                "Could not read the config file (should not happen because we just succeeded just before). {e}"
538            ));
539        }
540    };
541
542    #[cfg(feature = "macro_debug")]
543    eprintln!("[build monitor type]");
544    let monitor_type = if let Some(monitor_config) = copper_config.get_monitor_config() {
545        let monitor_type = parse_str::<Type>(monitor_config.get_type())
546            .expect("Could not transform the monitor type name into a Rust type.");
547        quote! { #monitor_type }
548    } else {
549        quote! { NoMonitor }
550    };
551
552    // This is common for all the mission as it will be inserted in the respective modules with their local CuTasks, CuStampedDataSet etc...
553    #[cfg(feature = "macro_debug")]
554    eprintln!("[build runtime field]");
555    // add that to a new field
556    let runtime_field: Field = if sim_mode {
557        parse_quote! {
558            copper_runtime: cu29::curuntime::CuRuntime<CuSimTasks, CuBridges, CuStampedDataSet, #monitor_type, #DEFAULT_CLNB>
559        }
560    } else {
561        parse_quote! {
562            copper_runtime: cu29::curuntime::CuRuntime<CuTasks, CuBridges, CuStampedDataSet, #monitor_type, #DEFAULT_CLNB>
563        }
564    };
565
566    #[cfg(feature = "macro_debug")]
567    eprintln!("[match struct anonymity]");
568    match &mut application_struct.fields {
569        Named(fields_named) => {
570            fields_named.named.push(runtime_field);
571        }
572        Unnamed(fields_unnamed) => {
573            fields_unnamed.unnamed.push(runtime_field);
574        }
575        Fields::Unit => {
576            panic!(
577                "This struct is a unit struct, it should have named or unnamed fields. use struct Something {{}} and not struct Something;"
578            )
579        }
580    };
581
582    let all_missions = copper_config.graphs.get_all_missions_graphs();
583    let mut all_missions_tokens = Vec::<proc_macro2::TokenStream>::new();
584    for (mission, graph) in &all_missions {
585        let mission_mod = parse_str::<Ident>(mission.as_str())
586            .expect("Could not make an identifier of the mission name");
587
588        #[cfg(feature = "macro_debug")]
589        eprintln!("[extract tasks ids & types]");
590        let task_specs = CuTaskSpecSet::from_graph(graph);
591
592        let culist_channel_usage = collect_bridge_channel_usage(graph);
593        let mut culist_bridge_specs =
594            build_bridge_specs(&copper_config, graph, &culist_channel_usage);
595        let (culist_plan, culist_exec_entities, culist_plan_to_original) =
596            match build_execution_plan(graph, &task_specs, &mut culist_bridge_specs) {
597                Ok(plan) => plan,
598                Err(e) => return return_error(format!("Could not compute copperlist plan: {e}")),
599            };
600        let task_member_names = collect_task_member_names(graph);
601        let (culist_call_order, node_output_positions) = collect_culist_metadata(
602            &culist_plan,
603            &culist_exec_entities,
604            &mut culist_bridge_specs,
605            &culist_plan_to_original,
606        );
607
608        #[cfg(feature = "macro_debug")]
609        {
610            eprintln!("[runtime plan for mission {mission}]");
611            eprintln!("{culist_plan:?}");
612        }
613
614        let culist_support: proc_macro2::TokenStream = gen_culist_support(
615            &culist_plan,
616            &culist_call_order,
617            &node_output_positions,
618            &task_member_names,
619        );
620
621        let bundle_specs = match build_bundle_specs(&copper_config, mission.as_str()) {
622            Ok(specs) => specs,
623            Err(e) => return return_error(e.to_string()),
624        };
625        let threadpool_bundle_index = if task_specs.background_flags.iter().any(|&flag| flag) {
626            match bundle_specs
627                .iter()
628                .position(|bundle| bundle.id == "threadpool")
629            {
630                Some(index) => Some(index),

3683                debug!("Task {}: Error during process: {}", #mission_mod::TASKS_IDS[#tid], &error);
3684                let decision = monitor.process_error(#tid, CuTaskState::Process, &error);
3685                match decision {
3686                    Decision::Abort => {
3687                        debug!("Process: ABORT decision from monitoring. Task '{}' errored out \
3688                                during process. Skipping the processing of CL {}.", #mission_mod::TASKS_IDS[#tid], clid);
3689                        monitor.process_copperlist(&#mission_mod::collect_metadata(&culist))?;
3690                        cl_manager.end_of_processing(clid)?;
3691                        return Ok(());
3692                    }
3693                    Decision::Ignore => {
3694                        debug!("Process: IGNORE decision from monitoring. Task '{}' errored out \
3695                                during process. The runtime will continue with a forced empty message.", #mission_mod::TASKS_IDS[#tid]);
3696                        let cumsg_output = &mut msgs.#output_culist_index;
3697                        #output_clear_payload
3698                    }
3699                    Decision::Shutdown => {
3700                        debug!("Process: SHUTDOWN decision from monitoring. Task '{}' errored out \
3701                                during process. The runtime cannot continue.", #mission_mod::TASKS_IDS[#tid]);
3702                        return Err(CuError::new_with_cause("Task errored out during process.", error));
3703                    }
3704                }
3705            };
3706
3707            let call_sim_callback = if sim_mode {
3708                quote! {
3709                    let doit = {
3710                        let cumsg_input = &#inputs_type;
3711                        let cumsg_output = &mut msgs.#output_culist_index;
3712                        let state = CuTaskCallbackState::Process(cumsg_input, cumsg_output);
3713                        let ovr = sim_callback(SimStep::#enum_name(state));
3714
3715                        if let SimOverride::Errored(reason) = ovr  {
3716                            let error: CuError = reason.into();
3717                            #monitoring_action
3718                            false
3719                        }
3720                        else {
3721                            ovr == SimOverride::ExecuteByRuntime
3722                        }
3723                    };
3724                }
3725            } else {
3726                quote! { let doit = true; }
3727            };
3728
3729            let logging_tokens = if !task_specs.logging_enabled[tid] {
3730                quote! {
3731                    let mut cumsg_output = &mut culist.msgs.0.#output_culist_index;
3732                    #output_clear_payload
3733                }
3734            } else {
3735                quote!()
3736            };
3737
3738            (
3739                quote! {
3740                    {
3741                        #comment_tokens
3742                        kf_manager.freeze_task(clid, &#task_instance)?;
3743                        #call_sim_callback
3744                        let cumsg_input = &#inputs_type;
3745                        let cumsg_output = &mut msgs.#output_culist_index;
3746                        #output_start_time
3747                        let maybe_error = if doit {
3748                            #rt_guard
3749                            #task_instance.process(clock, cumsg_input, cumsg_output)
3750                        } else {
3751                            Ok(())
3752                        };
3753                        #output_end_time
3754                        if let Err(error) = maybe_error {
3755                            #monitoring_action
3756                        }
3757                    }
3758                },
3759                logging_tokens,
3760            )
3761        }
3762    }

3780        .iter()

3790        quote! { &mut msgs.#culist_index_ts }

3792        let port_index = syn::Index::from(port_index);

3794    };

3796        channel
3797            .monitor_index
3798            .expect("Bridge Rx channel missing monitor index"),
3799    );
3800    let bridge_type = &bridge_spec.type_path;
3801    let const_ident = &channel.const_ident;
3802    (
3803        quote! {
3804            {
3805                let bridge = &mut bridges.#bridge_tuple_index;
3806                let cumsg_output = #output_ref;
3807                cumsg_output.metadata.process_time.start = clock.now().into();
3808                let maybe_error = {
3809                    #rt_guard
3810                    bridge.receive(
3811                        clock,
3812                        &<#bridge_type as cu29::cubridge::CuBridge>::Rx::#const_ident,

3819                    match decision {
3820                        Decision::Abort => {

3822                            monitor.process_copperlist(&#mission_mod::collect_metadata(&culist))?;
3823                            cl_manager.end_of_processing(clid)?;
3824                            return Ok(());
3825                        }
3826                        Decision::Ignore => {
3827                            debug!("Process: IGNORE decision from monitoring. Task '{}' errored out during process. The runtime will continue with a forced empty message.", #mission_mod::TASKS_IDS[#monitor_index]);
3828                            let cumsg_output = #output_ref;
3829                            cumsg_output.clear_payload();
3830                        }
3831                        Decision::Shutdown => {
3832                            debug!("Process: SHUTDOWN decision from monitoring. Task '{}' errored out during process. The runtime cannot continue.", #mission_mod::TASKS_IDS[#monitor_index]);
3833                            return Err(CuError::new_with_cause("Task errored out during process.", error));
3834                        }
3835                    }
3836                }
3837            }
3838        },
3839        quote! {},
