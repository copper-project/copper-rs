5fn is_struct_member_char(c: char) -> bool {
6    c.is_alphanumeric() || c == '_'
7}

9/// Small tool to create a valid enum entry from an identifier.
10pub(crate) fn config_id_to_enum(id: &str) -> String {
11    let mut candidate = id
12        .chars()
13        .map(|c| if c.is_alphanumeric() { c } else { '_' })
14        .collect::<String>();
15
16    candidate = candidate.to_case(Case::Pascal);
17
18    if candidate.chars().next().is_some_and(|c| c.is_ascii_digit()) {
19        candidate.insert(0, '_');
20    }
21
22    candidate
23}

25/// Same as config_id_to_enum but for a struct member name
26pub(crate) fn config_id_to_struct_member(id: &str) -> String {
27    let mut candidate = id
28        .trim_start_matches(|c| !is_struct_member_char(c))
29        .trim_end_matches(|c| !is_struct_member_char(c))
30        .chars()
31        .map(|c| if is_struct_member_char(c) { c } else { '_' })
32        .collect::<String>();
33
34    candidate = candidate.to_case(Case::Snake);
35
36    if candidate.is_empty() {
37        candidate.push('_');
38    }
39
40    if candidate.chars().next().is_some_and(|c| c.is_ascii_digit()) {
41        candidate.insert(0, '_');
42    }
43
44    candidate
45}

47/// Converts a configuration identifier into a SCREAMING_SNAKE_CASE name suitable
48/// for referencing bridge channel constants.
49pub(crate) fn config_id_to_bridge_const(id: &str) -> String {
50    let sanitized = id
51        .chars()
52        .map(|c| if c.is_alphanumeric() { c } else { '_' })
53        .collect::<String>();
54
55    // Replicate paste::paste! { [<$ident:snake:upper>] } behavior used by the tx/rx macros.
56    let mut snake = String::with_capacity(sanitized.len());
57    let mut prev = '_';
58    for ch in sanitized.chars() {
59        if ch.is_uppercase() && prev != '_' {
60            snake.push('_');
61        }
62        snake.push(ch);
63        prev = ch;
64    }
65
66    let mut candidate = snake.to_lowercase().to_uppercase();
67
68    if candidate.chars().next().is_some_and(|c| c.is_ascii_digit()) {
69        candidate.insert(0, '_');
70    }
71
72    candidate
73}

76pub fn caller_crate_root() -> PathBuf {
77    let crate_name =
78        std::env::var("CARGO_PKG_NAME").expect("failed to read ENV var `CARGO_PKG_NAME`!");
79    let current_dir = std::env::current_dir().expect("failed to unwrap env::current_dir()!");
80    let search_entry = format!("name=\"{crate_name}\"");
81    for entry in WalkDir::new(&current_dir)
82        .into_iter()
83        .filter_entry(|e| !e.file_name().eq_ignore_ascii_case("target"))
84    {
85        let Ok(entry) = entry else {
86            continue;
87        };
88        if !entry.file_type().is_file() {
89            continue;
90        }
91        let Some(file_name) = entry.path().file_name() else {
92            continue;
93        };
94        if !file_name.eq_ignore_ascii_case("Cargo.toml") {
95            continue;
96        }
97        let Ok(cargo_toml) = std::fs::read_to_string(entry.path()) else {
98            continue;
99        };
100        if cargo_toml
101            .chars()
102            .filter(|&c| !c.is_whitespace())
103            .collect::<String>()
104            .contains(search_entry.as_str())
105        {
106            return entry.path().parent().unwrap().to_path_buf();
107        }
108    }
109    current_dir
110}

116    fn is_valid_rust_identifier(input: &str) -> bool {
117        if input.is_empty() {
118            return false;
119        }
120
121        // Check if the first character is valid
122        let mut chars = input.chars();
123        if let Some(first) = chars.next()
124            && !first.is_alphabetic()
125            && first != '_'
126        {
127            return false;
128        }
129
130        // Check the rest of the characters
131        if !chars.all(|c| c.is_alphanumeric() || c == '_') {
132            return false;
133        }
134
135        // Check if it's a Rust keyword (can use a set of known keywords)
136        let keywords = [
137            "as", "break", "const", "continue", "crate", "else", "enum", "extern", "false", "fn",
138            "for", "if", "impl", "in", "let", "loop", "match", "mod", "move", "mut", "pub", "ref",
139            "return", "self", "Self", "static", "struct", "super", "trait", "true", "type",
140            "unsafe", "use", "where", "while",
141        ];
142
143        !keywords.contains(&input)
144    }
