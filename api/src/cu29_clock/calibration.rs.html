1use portable_atomic::{AtomicU64, Ordering};
2
3// Frequency estimation for converting raw counter-values to nanoseconds
4static FREQUENCY_NS: AtomicU64 = AtomicU64::new(0);
5static INIT_COUNTER: AtomicU64 = AtomicU64::new(0);
6static INIT_TIME_NS: AtomicU64 = AtomicU64::new(0);
7
8const CALIBRATION_PERIOD_NS: u64 = 10_000_000;
9
10/// Calibrates the high-precision clock vs. a real time clock
11pub fn calibrate(
12    read_raw_counter: fn() -> u64,
13    read_rtc_ns: impl Fn() -> u64 + Send + Sync + 'static,
14    sleep_ns: impl Fn(u64) + Send + Sync + 'static,
15) {
16    let start_counter = read_raw_counter();
17    let start_time = read_rtc_ns();
18
19    sleep_ns(CALIBRATION_PERIOD_NS);
20
21    let end_counter = read_raw_counter();
22    let end_time = read_rtc_ns();
23
24    let counter_diff = end_counter.saturating_sub(start_counter);
25    let time_diff_ns = end_time.saturating_sub(start_time);
26
27    if counter_diff > 0 {
28        let freq_ns = ((counter_diff as u128 * 1_000_000_000) / time_diff_ns as u128) as u64;
29        FREQUENCY_NS.store(freq_ns, Ordering::Relaxed);
30        INIT_COUNTER.store(start_counter, Ordering::Relaxed);
31        INIT_TIME_NS.store(start_time, Ordering::Relaxed);
32    }
33}
34
35/// Translate the high-precision counter-value to real time nanoseconds
36pub fn counter_to_nanos(read_raw_counter: fn() -> u64) -> u64 {
37    let freq = FREQUENCY_NS.load(Ordering::Relaxed);
38    let init_counter = INIT_COUNTER.load(Ordering::Relaxed);
39    let init_time_ns = INIT_TIME_NS.load(Ordering::Relaxed);
40    let counter = read_raw_counter();
41    let counter_diff = counter.saturating_sub(init_counter);
42
43    init_time_ns.saturating_add(((counter_diff as u128 * 1_000_000_000) / freq as u128) as u64)
44}
