124#[cfg(feature = "mcap")]
125fn run_cli_inner<P>() -> CuResult<()>
126where
127    P: CopperListTuple + mcap_export::PayloadSchemas,
128{
129    let args = LogReaderCli::parse();
130    let unifiedlog_base = args.unifiedlog_base;
131
132    let UnifiedLogger::Read(mut dl) = UnifiedLoggerBuilder::new()
133        .file_base_name(&unifiedlog_base)
134        .build()
135        .expect("Failed to create logger")
136    else {
137        panic!("Failed to create logger");
138    };
139
140    match args.command {
141        Command::ExtractTextLog { log_index } => {
142            let reader = UnifiedLoggerIOReader::new(dl, UnifiedLogType::StructuredLogLine);
143            textlog_dump(reader, &log_index)?;
144        }
145        Command::ExtractCopperlists { export_format } => {
146            println!("Extracting copperlists with format: {export_format}");
147            let mut reader = UnifiedLoggerIOReader::new(dl, UnifiedLogType::CopperList);
148            let iter = copperlists_reader::<P>(&mut reader);
149
150            match export_format {
151                ExportFormat::Json => {
152                    for entry in iter {
153                        serde_json::to_writer_pretty(std::io::stdout(), &entry).unwrap();
154                    }
155                }
156                ExportFormat::Csv => {
157                    let mut first = true;
158                    for origin in P::get_all_task_ids() {
159                        if !first {
160                            print!(", ");
161                        } else {
162                            print!("id, ");
163                        }
164                        print!("{origin}_time, {origin}_tov, {origin},");
165                        first = false;
166                    }
167                    println!();
168                    for entry in iter {
169                        let mut first = true;
170                        for msg in entry.cumsgs() {
171                            if let Some(payload) = msg.payload() {
172                                if !first {
173                                    print!(", ");
174                                } else {
175                                    print!("{}, ", entry.id);
176                                }
177                                let metadata = msg.metadata();
178                                print!("{}, {}, ", metadata.process_time(), msg.tov());
179                                serde_json::to_writer(std::io::stdout(), payload).unwrap(); // TODO: escape for CSV
180                                first = false;
181                            }
182                        }
183                        println!();
184                    }
185                }
186            }
187        }
188        Command::Fsck { verbose } => {
189            if let Some(value) = check::<P>(&mut dl, verbose) {
190                return value;
191            }
192        }
193        #[cfg(feature = "mcap")]
194        Command::ExportMcap {
195            output,
196            progress,
197            quiet,
198        } => {
199            println!("Exporting copperlists to MCAP format: {}", output.display());
200
201            let show_progress = should_show_progress(progress, quiet);
202            let total_bytes = if show_progress {
203                Some(copperlist_total_bytes(&unifiedlog_base)?)
204            } else {
205                None
206            };
207
208            let reader = UnifiedLoggerIOReader::new(dl, UnifiedLogType::CopperList);
209
210            // Export to MCAP with schemas
211            // Note: P must implement PayloadSchemas trait (generated by gen_cumsgs! macro with mcap feature)
212            let stats = if let Some(total_bytes) = total_bytes {
213                let progress_bar = make_progress_bar(total_bytes);
214                let reader = ProgressReader::new(reader, progress_bar.clone());
215                let result = export_to_mcap_impl::<P>(reader, &output);
216                progress_bar.finish_and_clear();
217                result?
218            } else {
219                export_to_mcap_impl::<P>(reader, &output)?
220            };
221            println!("{stats}");
222        }
223        #[cfg(feature = "mcap")]
224        Command::McapInfo {
225            mcap_file,
226            schemas,
227            sample_messages,
228        } => {
229            mcap_info(&mcap_file, schemas, sample_messages)?;
230        }
231    }
232
233    Ok(())
234}

334#[cfg(feature = "mcap")]
335impl<R: Read> Read for ProgressReader<R> {
336    fn read(&mut self, buf: &mut [u8]) -> std::io::Result<usize> {
337        let read = self.inner.read(buf)?;
338        if read > 0 {
339            self.progress.inc(read as u64);
340        }
341        Ok(read)
342    }

345#[cfg(feature = "mcap")]
346fn make_progress_bar(total_bytes: u64) -> ProgressBar {
347    let progress_bar = ProgressBar::new(total_bytes);
348    progress_bar.set_draw_target(ProgressDrawTarget::stderr_with_hz(5));
349
350    let style = ProgressStyle::with_template(
351        "[{elapsed_precise}] {bar:40} {bytes}/{total_bytes} ({bytes_per_sec}, ETA {eta})",
352    )
353    .unwrap_or_else(|_| ProgressStyle::default_bar());
354
355    progress_bar.set_style(style.progress_chars("=>-"));
356    progress_bar
357}

373fn read_next_entry<T: Decode<()>>(src: &mut impl Read) -> Option<T> {
374    let entry = decode_from_std_read::<T, _, _>(src, standard());
375    match entry {
376        Ok(entry) => Some(entry),
377        Err(DecodeError::UnexpectedEnd { .. }) => None,
378        Err(DecodeError::Io { inner, additional }) => {
379            if inner.kind() == std::io::ErrorKind::UnexpectedEof {
380                None
381            } else {
382                println!("Error {inner:?} additional:{additional}");
383                None
384            }
385        }
386        Err(e) => {
387            println!("Error {e:?}");
388            None
389        }
390    }
391}

393/// Extracts the copper lists from a binary representation.
394/// P is the Payload determined by the configuration of the application.
395pub fn copperlists_reader<P: CopperListTuple>(
396    mut src: impl Read,
397) -> impl Iterator<Item = CopperList<P>> {
398    std::iter::from_fn(move || read_next_entry::<CopperList<P>>(&mut src))
399}

401/// Extracts the keyframes from the log.
402pub fn keyframes_reader(mut src: impl Read) -> impl Iterator<Item = KeyFrame> {
403    std::iter::from_fn(move || read_next_entry::<KeyFrame>(&mut src))
404}

406pub fn structlog_reader(mut src: impl Read) -> impl Iterator<Item = CuResult<CuLogEntry>> {
407    std::iter::from_fn(move || {
408        let entry = decode_from_std_read::<CuLogEntry, _, _>(&mut src, standard());
409
410        match entry {
411            Err(DecodeError::UnexpectedEnd { .. }) => None,
412            Err(DecodeError::Io {
413                inner,
414                additional: _,
415            }) => {
416                if inner.kind() == std::io::ErrorKind::UnexpectedEof {
417                    None
418                } else {
419                    Some(Err(CuError::new_with_cause("Error reading log", inner)))
420                }
421            }
422            Err(e) => Some(Err(CuError::new_with_cause("Error reading log", e))),
423            Ok(entry) => {
424                if entry.msg_index == 0 {
425                    None
426                } else {
427                    Some(Ok(entry))
428                }
429            }
430        }
431    })
432}

434/// Full dump of the copper structured log from its binary representation.
435/// This rebuilds a textual log.
436/// src: the source of the log data
437/// index: the path to the index file (containing the interned strings constructed at build time)
438pub fn textlog_dump(src: impl Read, index: &Path) -> CuResult<()> {
439    let all_strings = read_interned_strings(index).map_err(|e| {
440        CuError::new_with_cause(
441            "Failed to read interned strings from index",
442            std::io::Error::other(e),
443        )
444    })?;
445
446    for result in structlog_reader(src) {
447        match result {
448            Ok(entry) => match rebuild_logline(&all_strings, &entry) {
449                Ok(line) => println!("{line}"),
450                Err(e) => println!("Failed to rebuild log line: {e:?}"),
451            },
452            Err(e) => return Err(e),
453        }
454    }
455
456    Ok(())
457}

551    /// This is a python wrapper for CuLogEntries.
552    #[pyclass]
553    pub struct PyCuLogEntry {
554        pub inner: CuLogEntry,
555    }
556
557    #[pymethods]
558    impl PyCuLogEntry {
559        /// Returns the timestamp of the log entry.
560        pub fn ts<'a>(&self, py: Python<'a>) -> Bound<'a, PyDelta> {
561            let nanoseconds: u64 = self.inner.time.into();
562
563            // Convert nanoseconds to seconds and microseconds
564            let days = (nanoseconds / 86_400_000_000_000) as i32;
565            let seconds = (nanoseconds / 1_000_000_000) as i32;
566            let microseconds = ((nanoseconds % 1_000_000_000) / 1_000) as i32;
567
568            PyDelta::new(py, days, seconds, microseconds, false).unwrap()
569        }
570
571        /// Returns the index of the message in the vector of interned strings.
572        pub fn msg_index(&self) -> u32 {
573            self.inner.msg_index
574        }
575
576        /// Returns the index of the parameter names in the vector of interned strings.
577        pub fn paramname_indexes(&self) -> Vec<u32> {
578            self.inner.paramname_indexes.iter().copied().collect()
579        }
580
581        /// Returns the parameters of this log line
582        pub fn params(&self) -> Vec<Py<PyAny>> {
583            self.inner.params.iter().map(value_to_py).collect()
584        }
585    }

597    fn value_to_py(value: &cu29::prelude::Value) -> Py<PyAny> {
598        match value {
599            Value::String(s) => Python::attach(|py| s.into_pyobject(py).unwrap().into()),
600            Value::U64(u) => Python::attach(|py| u.into_pyobject(py).unwrap().into()),
601            Value::I64(i) => Python::attach(|py| i.into_pyobject(py).unwrap().into()),
602            Value::F64(f) => Python::attach(|py| f.into_pyobject(py).unwrap().into()),
603            Value::Bool(b) => Python::attach(|py| b.into_pyobject(py).unwrap().to_owned().into()),
604            Value::CuTime(t) => Python::attach(|py| t.0.into_pyobject(py).unwrap().into()),
605            Value::Bytes(b) => Python::attach(|py| b.into_pyobject(py).unwrap().into()),
606            Value::Char(c) => Python::attach(|py| c.into_pyobject(py).unwrap().into()),
607            Value::I8(i) => Python::attach(|py| i.into_pyobject(py).unwrap().into()),
608            Value::U8(u) => Python::attach(|py| u.into_pyobject(py).unwrap().into()),
609            Value::I16(i) => Python::attach(|py| i.into_pyobject(py).unwrap().into()),
610            Value::U16(u) => Python::attach(|py| u.into_pyobject(py).unwrap().into()),
611            Value::I32(i) => Python::attach(|py| i.into_pyobject(py).unwrap().into()),
612            Value::U32(u) => Python::attach(|py| u.into_pyobject(py).unwrap().into()),
613            Value::Map(m) => Python::attach(|py| {
614                let dict = PyDict::new(py);
615                for (k, v) in m.iter() {
616                    dict.set_item(value_to_py(k), value_to_py(v)).unwrap();
617                }
618                dict.into_pyobject(py).unwrap().into()
619            }),
620            Value::F32(f) => Python::attach(|py| f.into_pyobject(py).unwrap().into()),
621            Value::Option(o) => Python::attach(|py| {
622                if o.is_none() {
623                    py.None()
624                } else {
625                    o.clone().map(|v| value_to_py(&v)).unwrap()
626                }
627            }),
628            Value::Unit => Python::attach(|py| py.None()),
629            Value::Newtype(v) => value_to_py(v),
630            Value::Seq(s) => Python::attach(|py| {
631                let list = PyList::new(py, s.iter().map(value_to_py)).unwrap();
632                list.into_pyobject(py).unwrap().into()
633            }),
634        }
635    }

749    /// Checks if we can recover the copper lists from a binary representation.
750    #[test]
751    fn test_copperlists_dump() {
752        let mut data = vec![0u8; 10000];
753        let mypls: [MyMsgs; 4] = [
754            MyMsgs((1, 2, 3.0)),
755            MyMsgs((2, 3, 4.0)),
756            MyMsgs((3, 4, 5.0)),
757            MyMsgs((4, 5, 6.0)),
758        ];
759
760        let mut offset: usize = 0;
761        for pl in mypls.iter() {
762            let cl = CopperList::<MyMsgs>::new(1, *pl);
763            offset +=
764                encode_into_slice(&cl, &mut data.as_mut_slice()[offset..], standard()).unwrap();
765        }
766
767        let reader = Cursor::new(data);
768
769        let mut iter = copperlists_reader::<MyMsgs>(reader);
770        assert_eq!(iter.next().unwrap().msgs, MyMsgs((1, 2, 3.0)));
771        assert_eq!(iter.next().unwrap().msgs, MyMsgs((2, 3, 4.0)));
772        assert_eq!(iter.next().unwrap().msgs, MyMsgs((3, 4, 5.0)));
773        assert_eq!(iter.next().unwrap().msgs, MyMsgs((4, 5, 6.0)));
774    }
