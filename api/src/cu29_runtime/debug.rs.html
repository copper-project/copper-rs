1//! CuDebug: lightweight time-travel debugger helpers on top of Copper logs.
2//!
3//! Design goals:
4//! - Do **not** load entire copperlists into memory (logs can be huge).
5//! - Build a compact section index in one streaming pass (copperlists + keyframes).
6//! - Keep keyframes in memory (much smaller) and lazily page copperlist sections
7//!   with a tiny LRU cache for snappy stepping.
8//! - Reuse the public `CuSimApplication` API and user-provided sim callbacks.
9

28/// Result of a jump/step, useful for benchmarking cache effectiveness.
29#[derive(Debug, Clone)]
30pub struct JumpOutcome {

32    pub culistid: u32,
33    /// Keyframe used to rewind (if any)

58/// A reusable debugging session that can time-travel within a recorded log.
59///
60/// `CB` builds a simulation callback for a specific copperlist entry. This keeps the
61/// API generic: the caller can replay recorded outputs, poke clocks, or inject extra
62/// assertions inside the callback. `TF` extracts a timestamp from a copperlist to
63/// support time-based seeking.
64const DEFAULT_SECTION_CACHE_CAP: usize = 8;
65pub struct CuDebugSession<App, P, CB, TF, S, L>
66where
67    P: CopperListTuple,
68    S: SectionStorage,
69    L: UnifiedLogWrite<S> + 'static,
70{
71    app: App,
72    robot_clock: RobotClock,
73    clock_mock: RobotClockMock,
74    log_reader: UnifiedLoggerRead,
75    sections: Vec<SectionIndexEntry>,
76    total_entries: usize,
77    keyframes: Vec<KeyFrame>,
78    started: bool,
79    current_idx: Option<usize>,
80    last_keyframe: Option<u32>,
81    build_callback: CB,
82    time_of: TF,
83    // Tiny LRU cache of decoded sections
84    cache: HashMap<usize, CachedSection<P>>,
85    cache_order: VecDeque<usize>,
86    cache_cap: usize,
87    phantom: PhantomData<(S, L)>,
88}

212    fn ensure_started(&mut self) -> CuResult<()> {
213        if self.started {
214            return Ok(());
215        }
216        let mut noop = |_step: App::Step<'_>| SimOverride::ExecuteByRuntime;
217        self.app.start_all_tasks(&mut noop)?;
218        self.started = true;
219        Ok(())
220    }

222    fn nearest_keyframe(&self, target_culistid: u32) -> Option<KeyFrame> {
223        self.keyframes
224            .iter()
225            .filter(|kf| kf.culistid <= target_culistid)
226            .max_by_key(|kf| kf.culistid)
227            .cloned()

230    fn restore_keyframe(&mut self, kf: &KeyFrame) -> CuResult<()> {
231        self.app.restore_keyframe(kf)?;
232        self.clock_mock.set_value(kf.timestamp.as_nanos());
233        self.last_keyframe = Some(kf.culistid);
234        Ok(())

237    fn find_section_for_index(&self, idx: usize) -> Option<usize> {
238        self.sections
239            .binary_search_by(|s| {
240                if idx < s.start_idx {
241                    std::cmp::Ordering::Greater
242                } else if idx >= s.start_idx + s.len {
243                    std::cmp::Ordering::Less
244                } else {
245                    std::cmp::Ordering::Equal
246                }
247            })
248            .ok()
249    }
250
251    fn find_section_for_culistid(&self, culistid: u32) -> Option<usize> {
252        self.sections
253            .binary_search_by(|s| {
254                if culistid < s.first_id {
255                    std::cmp::Ordering::Greater
256                } else if culistid > s.last_id {
257                    std::cmp::Ordering::Less
258                } else {
259                    std::cmp::Ordering::Equal
260                }
261            })
262            .ok()
263    }

265    /// Lower-bound lookup: return the first section whose `first_ts >= ts`.
266    /// If `ts` is earlier than the first section, return the first section.
267    /// Return `None` only when `ts` is beyond the last section's range.
268    fn find_section_for_time(&self, ts: CuTime) -> Option<usize> {
269        if self.sections.is_empty() {
270            return None;
271        }
272
273        // Fast path when all sections carry timestamps.
274        if self.sections.iter().all(|s| s.first_ts.is_some()) {
275            let idx = match self.sections.binary_search_by(|s| {
276                let a = s.first_ts.unwrap();
277                if a < ts {
278                    std::cmp::Ordering::Less
279                } else if a > ts {
280                    std::cmp::Ordering::Greater
281                } else {
282                    std::cmp::Ordering::Equal
283                }
284            }) {
285                Ok(i) => i,
286                Err(i) => i, // insertion point = first first_ts >= ts
287            };
288
289            if idx < self.sections.len() {
290                return Some(idx);
291            }
292
293            // ts is after the last section start; allow selecting the last section
294            // if the timestamp still lies inside its recorded range.
295            let last = self.sections.last().unwrap();
296            if let Some(last_ts) = last.last_ts
297                && ts <= last_ts
298            {
299                return Some(self.sections.len() - 1);
300            }
301            return None;
302        }
303
304        // Fallback for sections missing timestamps: choose first window that contains ts;
305        // if ts is earlier than the first timestamped section, pick that section; otherwise
306        // only return None when ts is past the last known range.
307        if let Some(first_ts) = self.sections.first().and_then(|s| s.first_ts)
308            && ts <= first_ts
309        {
310            return Some(0);
311        }
312
313        if let Some(idx) = self
314            .sections
315            .iter()
316            .position(|s| match (s.first_ts, s.last_ts) {
317                (Some(a), Some(b)) => a <= ts && ts <= b,
318                (Some(a), None) => a <= ts,
319                _ => false,
320            })
321        {
322            return Some(idx);
323        }
324
325        let last = self.sections.last().unwrap();
326        match last.last_ts {
327            Some(b) if ts <= b => Some(self.sections.len() - 1),
328            _ => None,
329        }
330    }

332    fn touch_cache(&mut self, key: usize) {
333        if let Some(pos) = self.cache_order.iter().position(|k| *k == key) {
334            self.cache_order.remove(pos);
335        }
336        self.cache_order.push_back(key);
337        while self.cache_order.len() > self.cache_cap {
338            if let Some(old) = self.cache_order.pop_front() {
339                self.cache.remove(&old);
340            }
341        }
342    }

344    fn load_section(&mut self, section_idx: usize) -> CuResult<&CachedSection<P>> {
345        if self.cache.contains_key(&section_idx) {
346            self.touch_cache(section_idx);
347            // SAFETY: key exists, unwrap ok.
348            return Ok(self.cache.get(&section_idx).unwrap());
349        }
350
351        let entry = &self.sections[section_idx];
352        let (header, data) = read_section_at(&mut self.log_reader, entry.pos)?;
353        if header.entry_type != UnifiedLogType::CopperList {
354            return Err(CuError::from(
355                "Section type mismatch while loading copperlists",
356            ));
357        }
358
359        let (entries, timestamps) = decode_copperlists::<P, _>(&data, &self.time_of)?;
360        let cached = CachedSection {
361            entries,
362            timestamps,
363        };
364        self.cache.insert(section_idx, cached);
365        self.touch_cache(section_idx);
366        Ok(self.cache.get(&section_idx).unwrap())
367    }

369    fn copperlist_at(
370        &mut self,
371        idx: usize,
372    ) -> CuResult<(Arc<crate::copperlist::CopperList<P>>, Option<CuTime>)> {
373        let section_idx = self
374            .find_section_for_index(idx)
375            .ok_or_else(|| CuError::from("Index outside copperlist log"))?;
376        let start_idx = self.sections[section_idx].start_idx;
377        let section = self.load_section(section_idx)?;
378        let local = idx - start_idx;
379        let cl = section
380            .entries
381            .get(local)
382            .ok_or_else(|| CuError::from("Corrupt section index vs cache"))?
383            .clone();
384        let ts = section.timestamps.get(local).copied().unwrap_or(None);
385        Ok((cl, ts))
386    }

388    fn index_for_culistid(&mut self, culistid: u32) -> CuResult<usize> {
389        let section_idx = self
390            .find_section_for_culistid(culistid)
391            .ok_or_else(|| CuError::from("Requested culistid not present in log"))?;
392        let start_idx = self.sections[section_idx].start_idx;
393        let section = self.load_section(section_idx)?;
394        let mut idx = start_idx;
395        for cl in &section.entries {
396            if cl.id == culistid {
397                return Ok(idx);
398            }
399            idx += 1;
400        }
401        Err(CuError::from("culistid not found inside indexed section"))
402    }
403
404    fn index_for_time(&mut self, ts: CuTime) -> CuResult<usize> {
405        let section_idx = self
406            .find_section_for_time(ts)
407            .ok_or_else(|| CuError::from("No copperlist at or after requested timestamp"))?;
408        let start_idx = self.sections[section_idx].start_idx;
409        let section = self.load_section(section_idx)?;
410        let idx = start_idx;
411        for (i, maybe) in section.timestamps.iter().enumerate() {
412            if matches!(maybe, Some(t) if *t >= ts) {
413                return Ok(idx + i);
414            }
415        }
416        Err(CuError::from("Timestamp not found within section"))
417    }

419    fn replay_range(&mut self, start: usize, end: usize) -> CuResult<usize> {
420        let mut replayed = 0usize;
421        for idx in start..=end {
422            let (entry, ts) = self.copperlist_at(idx)?;
423            if let Some(ts) = ts {
424                self.clock_mock.set_value(ts.as_nanos());
425            }
426            let clock_for_cb = self.robot_clock.clone();
427            let mut cb = (self.build_callback)(entry.as_ref(), clock_for_cb);
428            self.app.run_one_iteration(&mut cb)?;
429            replayed += 1;
430            self.current_idx = Some(idx);
431        }
432        Ok(replayed)
433    }

435    fn goto_index(&mut self, target_idx: usize) -> CuResult<JumpOutcome> {
436        self.ensure_started()?;
437        if target_idx >= self.total_entries {
438            return Err(CuError::from("Target index outside log"));
439        }
440        let (target_cl, _) = self.copperlist_at(target_idx)?;
441        let target_culistid = target_cl.id;
442
443        let keyframe_used: Option<u32>;
444        let replay_start: usize;
445
446        // Fast path: forward stepping from current state.
447        if let Some(current) = self.current_idx {
448            if target_idx >= current {
449                replay_start = current + 1;
450                keyframe_used = self.last_keyframe;
451            } else {
452                // Need to rewind to nearest keyframe
453                let Some(kf) = self.nearest_keyframe(target_culistid) else {
454                    return Err(CuError::from("No keyframe available to rewind"));
455                };
456                self.restore_keyframe(&kf)?;
457                keyframe_used = Some(kf.culistid);
458                replay_start = self.index_for_culistid(kf.culistid)?;
459            }
460        } else {
461            // First jump: align to nearest keyframe
462            let Some(kf) = self.nearest_keyframe(target_culistid) else {
463                return Err(CuError::from("No keyframe found in log"));
464            };
465            self.restore_keyframe(&kf)?;
466            keyframe_used = Some(kf.culistid);
467            replay_start = self.index_for_culistid(kf.culistid)?;
468        }
469
470        if replay_start > target_idx {
471            return Err(CuError::from(
472                "Replay start past target index; log ordering issue",
473            ));
474        }
475
476        let replayed = self.replay_range(replay_start, target_idx)?;
477
478        Ok(JumpOutcome {
479            culistid: target_culistid,
480            keyframe_culistid: keyframe_used,
481            replayed,
482        })
483    }

497    /// Step relative to the current cursor. Negative values rewind via keyframe.
498    pub fn step(&mut self, delta: i32) -> CuResult<JumpOutcome> {
499        let current =
500            self.current_idx
501                .ok_or_else(|| CuError::from("Cannot step before any jump"))? as i32;
502        let target = current + delta;
503        if target < 0 || target as usize >= self.total_entries {
504            return Err(CuError::from("Step would move outside log bounds"));
505        }
506        self.goto_index(target as usize)
507    }

531/// Decode all copperlists contained in a single unified-log section.
532#[allow(clippy::type_complexity)]
533fn decode_copperlists<
534    P: CopperListTuple,
535    TF: Fn(&crate::copperlist::CopperList<P>) -> Option<CuTime>,
536>(
537    section: &[u8],
538    time_of: &TF,
539) -> CuResult<(
540    Vec<Arc<crate::copperlist::CopperList<P>>>,
541    Vec<Option<CuTime>>,
542)> {
543    let mut cursor = std::io::Cursor::new(section);
544    let mut entries = Vec::new();
545    let mut timestamps = Vec::new();
546    loop {
547        match decode_from_std_read::<crate::copperlist::CopperList<P>, _, _>(
548            &mut cursor,
549            standard(),
550        ) {
551            Ok(cl) => {
552                timestamps.push(time_of(&cl));
553                entries.push(Arc::new(cl));
554            }
555            Err(DecodeError::UnexpectedEnd { .. }) => break,
556            Err(DecodeError::Io { inner, .. }) if inner.kind() == io::ErrorKind::UnexpectedEof => {
557                break;
558            }
559            Err(e) => {
560                return Err(CuError::new_with_cause(
561                    "Failed to decode CopperList section",
562                    e,
563                ));
564            }
565        }
566    }
567    Ok((entries, timestamps))
568}

570/// Scan a copperlist section for metadata only.
571#[allow(clippy::type_complexity)]
572fn scan_copperlist_section<
573    P: CopperListTuple,
574    TF: Fn(&crate::copperlist::CopperList<P>) -> Option<CuTime>,
575>(
576    section: &[u8],
577    time_of: &TF,
578) -> CuResult<(usize, u32, u32, Option<CuTime>, Option<CuTime>)> {
579    let mut cursor = std::io::Cursor::new(section);
580    let mut count = 0usize;
581    let mut first_id = None;
582    let mut last_id = None;
583    let mut first_ts = None;
584    let mut last_ts = None;
585    loop {
586        match decode_from_std_read::<crate::copperlist::CopperList<P>, _, _>(
587            &mut cursor,
588            standard(),
589        ) {
590            Ok(cl) => {
591                let ts = time_of(&cl);
592                if ts.is_none() {
593                    #[cfg(feature = "std")]
594                    eprintln!(
595                        "CuDebug index warning: missing timestamp on culistid {}; time-based seek may be less accurate",
596                        cl.id
597                    );
598                }
599                if first_id.is_none() {
600                    first_id = Some(cl.id);
601                    first_ts = ts;
602                }
603                // Recover first_ts if the first entry lacked a timestamp but a later one has it.
604                if first_ts.is_none() {
605                    first_ts = ts;
606                }
607                last_id = Some(cl.id);
608                last_ts = ts.or(last_ts);
609                count += 1;
610            }
611            Err(DecodeError::UnexpectedEnd { .. }) => break,
612            Err(DecodeError::Io { inner, .. }) if inner.kind() == io::ErrorKind::UnexpectedEof => {
613                break;
614            }
615            Err(e) => {
616                return Err(CuError::new_with_cause(
617                    "Failed to scan copperlist section",
618                    e,
619                ));
620            }
621        }
622    }
623    let first_id = first_id.ok_or_else(|| CuError::from("Empty copperlist section"))?;
624    let last_id = last_id.unwrap_or(first_id);
625    Ok((count, first_id, last_id, first_ts, last_ts))
626}

628/// Build a reusable read-only unified logger for this session.
629fn build_read_logger(log_base: &Path) -> CuResult<UnifiedLoggerRead> {
630    let logger = UnifiedLoggerBuilder::new()
631        .file_base_name(log_base)
632        .build()
633        .map_err(|e| CuError::new_with_cause("Failed to open unified log", e))?;
634    let UnifiedLogger::Read(dl) = logger else {
635        return Err(CuError::from("Expected read-only unified logger"));
636    };
637    Ok(dl)
638}

649/// Build a section-level index in one pass (copperlists + keyframes).
650fn index_log<P, TF>(
651    log_base: &Path,
652    time_of: &TF,
653) -> CuResult<(Vec<SectionIndexEntry>, Vec<KeyFrame>, usize)>
654where
655    P: CopperListTuple,
656    TF: Fn(&crate::copperlist::CopperList<P>) -> Option<CuTime>,
657{
658    let logger = UnifiedLoggerBuilder::new()
659        .file_base_name(log_base)
660        .build()
661        .map_err(|e| CuError::new_with_cause("Failed to open unified log", e))?;
662    let UnifiedLogger::Read(mut dl) = logger else {
663        return Err(CuError::from("Expected read-only unified logger"));
664    };
665
666    let mut sections = Vec::new();
667    let mut keyframes = Vec::new();
668    let mut total_entries = 0usize;
669
670    loop {
671        let pos = dl.position();
672        let (header, data) = dl.raw_read_section()?;
673        if header.entry_type == UnifiedLogType::LastEntry {
674            break;
675        }
676
677        match header.entry_type {
678            UnifiedLogType::CopperList => {
679                let (len, first_id, last_id, first_ts, last_ts) =
680                    scan_copperlist_section::<P, _>(&data, time_of)?;
681                if len == 0 {
682                    continue;
683                }
684                sections.push(SectionIndexEntry {
685                    pos,
686                    start_idx: total_entries,
687                    len,
688                    first_id,
689                    last_id,
690                    first_ts,
691                    last_ts,
692                });
693                total_entries += len;
694            }
695            UnifiedLogType::FrozenTasks => {
696                // Read all keyframes in this section
697                let mut cursor = std::io::Cursor::new(&data);
698                loop {
699                    match decode_from_std_read::<KeyFrame, _, _>(&mut cursor, standard()) {
700                        Ok(kf) => keyframes.push(kf),
701                        Err(DecodeError::UnexpectedEnd { .. }) => break,
702                        Err(DecodeError::Io { inner, .. })
703                            if inner.kind() == io::ErrorKind::UnexpectedEof =>
704                        {
705                            break;
706                        }
707                        Err(e) => {
708                            return Err(CuError::new_with_cause(
709                                "Failed to decode keyframe section",
710                                e,
711                            ));
712                        }
713                    }
714                }
715            }
716            _ => {
717                // ignore other sections
718            }
719        }
720    }
721
722    Ok((sections, keyframes, total_entries))
723}
