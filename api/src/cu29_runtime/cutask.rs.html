21/// The state of a task.
22// Everything that is stateful in copper for zero copy constraints need to be restricted to this trait.
23pub trait CuMsgPayload:
24    Default + Debug + Clone + Encode + Decode<()> + Serialize + DeserializeOwned + Sized
25{
26}

28pub trait CuMsgPack {}

36macro_rules! impl_cu_msg_pack {
37    ($($name:ident),+) => {
38        impl<'cl, $($name),+> CuMsgPack for ($(&CuMsg<$name>,)+)
39        where
40            $($name: CuMsgPayload),+
41        {}
42    };
43}

46impl<T: CuMsgPayload> CuMsgPack for &CuMsg<T> {}

88/// CuMsgMetadata is a structure that contains metadata common to all CuStampedDataSet.
89#[derive(Debug, Clone, bincode::Encode, bincode::Decode, Serialize, Deserialize)]
90pub struct CuMsgMetadata {
91    /// The time range used for the processing of this message
92    pub process_time: PartialCuTimeRange,
93    /// A small string for real time feedback purposes.
94    /// This is useful for to display on the field when the tasks are operating correctly.
95    pub status_txt: CuCompactString,
96}
97
98impl Metadata for CuMsgMetadata {}
99
100impl CuMsgMetadata {
101    pub fn set_status(&mut self, status: impl ToCompactString) {
102        self.status_txt = CuCompactString(status.to_compact_string());
103    }
104}

107    fn process_time(&self) -> PartialCuTimeRange {
108        self.process_time
109    }
110
111    fn status_txt(&self) -> &CuCompactString {
112        &self.status_txt
113    }
114}

118        write!(

121            self.process_time.start, self.process_time.end

126/// CuMsg is the envelope holding the msg payload and the metadata between tasks.
127#[derive(Default, Debug, Clone, bincode::Encode, bincode::Decode, Serialize, Deserialize)]
128#[serde(bound(
129    serialize = "T: Serialize, M: Serialize",
130    deserialize = "T: DeserializeOwned, M: DeserializeOwned"
131))]
132pub struct CuStampedData<T, M>
133where
134    T: CuMsgPayload,
135    M: Metadata,
136{
137    /// This payload is the actual data exchanged between tasks.
138    payload: Option<T>,
139
140    /// The time of validity of the message.
141    /// It can be undefined (None), one measure point or a range of measures (TimeRange).
142    pub tov: Tov,
143
144    /// This metadata is the data that is common to all messages.
145    pub metadata: M,
146}
147
148impl Default for CuMsgMetadata {
149    fn default() -> Self {
150        CuMsgMetadata {
151            process_time: PartialCuTimeRange::default(),
152            status_txt: CuCompactString(CompactString::with_capacity(COMPACT_STRING_CAPACITY)),
153        }
154    }
155}

158where
159    T: CuMsgPayload,
160    M: Metadata,
161{
162    pub fn new(payload: Option<T>) -> Self {
163        CuStampedData {

165            tov: Tov::default(),
166            metadata: M::default(),
167        }
168    }
169    pub fn payload(&self) -> Option<&T> {
170        self.payload.as_ref()
171    }

173    pub fn set_payload(&mut self, payload: T) {
174        self.payload = Some(payload);
175    }
176
177    pub fn clear_payload(&mut self) {
178        self.payload = None;
179    }
180
181    pub fn payload_mut(&mut self) -> &mut Option<T> {

183    }
184}

206/// This is the robotics message type for Copper with the correct Metadata type
207/// that will be used by the runtime.
208pub type CuMsg<T> = CuStampedData<T, CuMsgMetadata>;
209
210impl<T: CuMsgPayload> CuStampedData<T, CuMsgMetadata> {
211    /// Reinterprets the payload type carried by this message.
212    ///
213    /// # Safety
214    ///
215    /// The caller must guarantee that the message really contains a payload of type `U`. Failing
216    /// to do so is undefined behaviour.
217    pub unsafe fn assume_payload<U: CuMsgPayload>(&self) -> &CuMsg<U> {
218        // SAFETY: Caller guarantees that the underlying payload is of type U.
219        unsafe { &*(self as *const CuMsg<T> as *const CuMsg<U>) }
220    }
221
222    /// Mutable variant of [`assume_payload`](Self::assume_payload).
223    ///
224    /// # Safety
225    ///
226    /// The caller must guarantee that mutating the returned message is sound for the actual
227    /// payload type stored in the buffer.
228    pub unsafe fn assume_payload_mut<U: CuMsgPayload>(&mut self) -> &mut CuMsg<U> {
229        // SAFETY: Caller guarantees that the underlying payload is of type U.
230        unsafe { &mut *(self as *mut CuMsg<T> as *mut CuMsg<U>) }
231    }
232}
233
234impl<T: CuMsgPayload + 'static> CuStampedData<T, CuMsgMetadata> {
235    fn downcast_err<U: CuMsgPayload + 'static>() -> CuError {
236        CuError::from(format!(
237            "CuMsg payload mismatch: {} cannot be reinterpreted as {}",
238            type_name::<T>(),
239            type_name::<U>()
240        ))
241    }
242
243    /// Attempts to view this message as carrying payload `U`.
244    pub fn downcast_ref<U: CuMsgPayload + 'static>(&self) -> CuResult<&CuMsg<U>> {
245        if TypeId::of::<T>() == TypeId::of::<U>() {
246            // SAFETY: We just proved that T == U.
247            Ok(unsafe { self.assume_payload::<U>() })
248        } else {
249            Err(Self::downcast_err::<U>())
250        }
251    }
252
253    /// Mutable variant of [`downcast_ref`](Self::downcast_ref).
254    pub fn downcast_mut<U: CuMsgPayload + 'static>(&mut self) -> CuResult<&mut CuMsg<U>> {
255        if TypeId::of::<T>() == TypeId::of::<U>() {
256            // SAFETY: We just proved that T == U.
257            Ok(unsafe { self.assume_payload_mut::<U>() })
258        } else {
259            Err(Self::downcast_err::<U>())
260        }
261    }
262}

264/// The internal state of a task needs to be serializable
265/// so the framework can take a snapshot of the task graph.
266pub trait Freezable {
267    /// This method is called by the framework when it wants to save the task state.
268    /// The default implementation is to encode nothing (stateless).
269    /// If you have a state, you need to implement this method.
270    fn freeze<E: Encoder>(&self, encoder: &mut E) -> Result<(), EncodeError> {
271        Encode::encode(&(), encoder) // default is stateless
272    }
273
274    /// This method is called by the framework when it wants to restore the task to a specific state.
275    /// Here it is similar to Decode but the framework will give you a new instance of the task (the new method will be called)
276    fn thaw<D: Decoder>(&mut self, _decoder: &mut D) -> Result<(), DecodeError> {
277        Ok(())
278    }
279}

281/// Bincode Adapter for Freezable tasks
282/// This allows the use of the bincode API directly to freeze and thaw tasks.
283pub struct BincodeAdapter<'a, T: Freezable + ?Sized>(pub &'a T);
284
285impl<'a, T: Freezable + ?Sized> Encode for BincodeAdapter<'a, T> {
286    fn encode<E: Encoder>(&self, encoder: &mut E) -> Result<(), EncodeError> {
287        self.0.freeze(encoder)
288    }
289}

291/// A Src Task is a task that only produces messages. For example drivers for sensors are Src Tasks.
292/// They are in push mode from the runtime.
293/// To set the frequency of the pulls and align them to any hw, see the runtime configuration.
294/// Note: A source has the privilege to have a clock passed to it vs a frozen clock.
295pub trait CuSrcTask: Freezable {
296    type Output<'m>: CuMsgPayload;
297    /// Resources required by the task.
298    type Resources<'r>;
299
300    /// Here you need to initialize everything your task will need for the duration of its lifetime.
301    /// The config allows you to access the configuration of the task.
302    fn new(_config: Option<&ComponentConfig>, _resources: Self::Resources<'_>) -> CuResult<Self>
303    where
304        Self: Sized;
305
306    /// Start is called between the creation of the task and the first call to pre/process.
307    fn start(&mut self, _clock: &RobotClock) -> CuResult<()> {
308        Ok(())
309    }
310
311    /// This is a method called by the runtime before "process". This is a kind of best effort,
312    /// as soon as possible call to give a chance for the task to do some work before to prepare
313    /// to make "process" as short as possible.
314    fn preprocess(&mut self, _clock: &RobotClock) -> CuResult<()> {
315        Ok(())
316    }
317
318    /// Process is the most critical execution of the task.
319    /// The goal will be to produce the output message as soon as possible.
320    /// Use preprocess to prepare the task to make this method as short as possible.
321    fn process<'o>(&mut self, clock: &RobotClock, new_msg: &mut Self::Output<'o>) -> CuResult<()>;
322
323    /// This is a method called by the runtime after "process". It is best effort a chance for
324    /// the task to update some state after process is out of the way.
325    /// It can be use for example to maintain statistics etc. that are not time-critical for the robot.
326    fn postprocess(&mut self, _clock: &RobotClock) -> CuResult<()> {
327        Ok(())
328    }
329
330    /// Called to stop the task. It signals that the *process method won't be called until start is called again.
331    fn stop(&mut self, _clock: &RobotClock) -> CuResult<()> {
332        Ok(())
333    }
334}
