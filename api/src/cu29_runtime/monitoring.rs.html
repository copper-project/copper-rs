152fn collect_output_ports(graph: &CuGraph, node_id: NodeId) -> Vec<(String, String)> {
153    let mut edge_ids = graph.get_src_edges(node_id).unwrap_or_default();
154    edge_ids.sort();
155
156    let mut outputs = Vec::new();
157    let mut seen = Vec::new();
158    let mut port_idx = 0usize;
159    for edge_id in edge_ids {
160        let Some(edge) = graph.edge(edge_id) else {
161            continue;
162        };
163        if seen.iter().any(|msg| msg == &edge.msg) {
164            continue;
165        }
166        seen.push(edge.msg.clone());
167        let mut port_label = String::from("out");
168        port_label.push_str(&port_idx.to_string());
169        port_label.push_str(": ");
170        port_label.push_str(edge.msg.as_str());
171        outputs.push((edge.msg.clone(), port_label));
172        port_idx += 1;
173    }
174    outputs
175}

177/// Derive a monitor-friendly topology from the runtime configuration.
178pub fn build_monitor_topology(
179    config: &CuConfig,
180    mission: Option<&str>,
181) -> CuResult<MonitorTopology> {
182    let graph = config.get_graph(mission)?;
183    let mut nodes: Map<String, MonitorNode> = Map::new();
184    let mut io_usage: Map<String, NodeIoUsage> = Map::new();
185    let mut output_port_lookup: Map<String, Map<String, String>> = Map::new();
186
187    let mut bridge_lookup: Map<&str, &BridgeConfig> = Map::new();
188    for bridge in &config.bridges {
189        bridge_lookup.insert(bridge.id.as_str(), bridge);
190    }
191
192    for cnx in graph.edges() {
193        io_usage.entry(cnx.src.clone()).or_default().has_outgoing = true;
194        io_usage.entry(cnx.dst.clone()).or_default().has_incoming = true;
195    }
196
197    for (_, node) in graph.get_all_nodes() {
198        let kind = match node.get_flavor() {
199            Flavor::Bridge => ComponentKind::Bridge,
200            _ => ComponentKind::Task,
201        };
202        let node_id = node.get_id();
203
204        let mut inputs = Vec::new();
205        let mut outputs = Vec::new();
206        if kind == ComponentKind::Bridge {
207            if let Some(bridge) = bridge_lookup.get(node_id.as_str()) {
208                for ch in &bridge.channels {
209                    match ch {
210                        BridgeChannelConfigRepresentation::Rx { id, .. } => {
211                            outputs.push(id.clone())
212                        }
213                        BridgeChannelConfigRepresentation::Tx { id, .. } => inputs.push(id.clone()),
214                    }
215                }
216            }
217        } else {
218            let usage = io_usage.get(node_id.as_str()).cloned().unwrap_or_default();
219            if usage.has_incoming || !usage.has_outgoing {
220                inputs.push("in".to_string());
221            }
222            if usage.has_outgoing {
223                if let Some(node_idx) = graph.get_node_id_by_name(node_id.as_str()) {
224                    let ports = collect_output_ports(graph, node_idx);
225                    let mut port_map: Map<String, String> = Map::new();
226                    for (msg_type, label) in ports {
227                        port_map.insert(msg_type, label.clone());
228                        outputs.push(label);
229                    }
230                    output_port_lookup.insert(node_id.clone(), port_map);
231                }
232            } else if !usage.has_incoming {
233                outputs.push("out".to_string());
234            }
235        }
236
237        nodes.insert(
238            node_id.clone(),
239            MonitorNode {
240                id: node_id,
241                type_name: Some(node.get_type().to_string()),
242                kind,
243                inputs,
244                outputs,
245            },
246        );
247    }
248
249    let mut connections = Vec::new();
250    for cnx in graph.edges() {
251        let src = cnx.src.clone();
252        let dst = cnx.dst.clone();
253
254        let src_port = cnx.src_channel.clone().or_else(|| {
255            output_port_lookup
256                .get(&src)
257                .and_then(|ports| ports.get(&cnx.msg).cloned())
258                .or_else(|| {
259                    nodes
260                        .get(&src)
261                        .and_then(|node| node.outputs.first().cloned())
262                })
263        });
264        let dst_port = cnx.dst_channel.clone().or_else(|| {
265            nodes
266                .get(&dst)
267                .and_then(|node| node.inputs.first().cloned())
268        });
269
270        connections.push(MonitorConnection {
271            src,
272            src_port,
273            dst,
274            dst_port,
275            msg: cnx.msg.clone(),
276        });
277    }
278
279    Ok(MonitorTopology {
280        nodes: nodes.into_values().collect(),
281        connections,
282    })
283}

333/// A simple allocator that counts the number of bytes allocated and deallocated.
334pub struct CountingAlloc<A: GlobalAlloc> {
335    inner: A,
336    allocated: AtomicUsize,
337    deallocated: AtomicUsize,
338}
339
340impl<A: GlobalAlloc> CountingAlloc<A> {
341    pub const fn new(inner: A) -> Self {
342        CountingAlloc {
343            inner,
344            allocated: AtomicUsize::new(0),
345            deallocated: AtomicUsize::new(0),
346        }
347    }
348
349    pub fn allocated(&self) -> usize {
350        self.allocated.load(Ordering::SeqCst)
351    }
352
353    pub fn deallocated(&self) -> usize {
354        self.deallocated.load(Ordering::SeqCst)
355    }
356
357    pub fn reset(&self) {
358        self.allocated.store(0, Ordering::SeqCst);
359        self.deallocated.store(0, Ordering::SeqCst);
360    }
361}
362
363// SAFETY: Delegates allocation/deallocation to the inner allocator while tracking sizes.
364unsafe impl<A: GlobalAlloc> GlobalAlloc for CountingAlloc<A> {
365    // SAFETY: Callers uphold the GlobalAlloc contract; we delegate to the inner allocator.
366    unsafe fn alloc(&self, layout: Layout) -> *mut u8 {
367        // SAFETY: Forwarding to the inner allocator preserves GlobalAlloc invariants.
368        let p = unsafe { self.inner.alloc(layout) };
369        if !p.is_null() {
370            self.allocated.fetch_add(layout.size(), Ordering::SeqCst);
371        }
372        p
373    }
374
375    // SAFETY: Callers uphold the GlobalAlloc contract; we delegate to the inner allocator.
376    unsafe fn dealloc(&self, ptr: *mut u8, layout: Layout) {
377        // SAFETY: Forwarding to the inner allocator preserves GlobalAlloc invariants.
378        unsafe { self.inner.dealloc(ptr, layout) }
379        self.deallocated.fetch_add(layout.size(), Ordering::SeqCst);
380    }
381}

387    bf_deallocated: usize,

397#[cfg(feature = "memory_monitoring")]
398impl ScopedAllocCounter {

400        ScopedAllocCounter {
401            bf_allocated: GLOBAL.allocated(),
402            bf_deallocated: GLOBAL.deallocated(),
403        }
404    }

406    /// Returns the total number of bytes allocated in the current scope
407    /// since the creation of this `ScopedAllocCounter`.
408    ///
409    /// # Example
410    /// ```
411    /// use cu29_runtime::monitoring::ScopedAllocCounter;
412    ///
413    /// let counter = ScopedAllocCounter::new();
414    /// let _vec = vec![0u8; 1024];

418        GLOBAL.allocated() - self.bf_allocated

422    /// since the creation of this `ScopedAllocCounter`.
423    ///
424    /// # Example
425    /// ```
426    /// use cu29_runtime::monitoring::ScopedAllocCounter;
427    ///
428    /// let counter = ScopedAllocCounter::new();
429    /// let _vec = vec![0u8; 1024];
430    /// drop(_vec);
431    /// println!("Bytes deallocated: {}", counter.get_deallocated());
432    /// ```
433    pub fn deallocated(&self) -> usize {
434        GLOBAL.deallocated() - self.bf_deallocated
435    }
436}

453#[cfg(feature = "std")]
454const BUCKET_COUNT: usize = 1024;
455#[cfg(not(feature = "std"))]
456const BUCKET_COUNT: usize = 256;

458/// Accumulative stat object that can give your some real time statistics.
459/// Uses a fixed-size bucketed histogram for accurate percentile calculations.
460#[derive(Debug, Clone)]
461pub struct LiveStatistics {
462    buckets: [u64; BUCKET_COUNT],
463    min_val: u64,
464    max_val: u64,
465    sum: u64,
466    sum_sq: u64,
467    count: u64,
468    max_value: u64,
469}
470
471impl LiveStatistics {
472    /// Creates a new `LiveStatistics` instance with a specified maximum value.
473    ///
474    /// This function initializes a `LiveStatistics` structure with default values
475    /// for tracking statistical data, while setting an upper limit for the data
476    /// points that the structure tracks.
477    ///
478    /// # Parameters
479    /// - `max_value` (`u64`): The maximum value that can be recorded or tracked.
480    ///
481    /// # Returns
482    /// A new instance of `LiveStatistics` with:
483    /// - `buckets`: An array pre-filled with zeros to categorize data points.
484    /// - `min_val`: Initialized to the maximum possible `u64` value to track the minimum correctly.
485    /// - `max_val`: Initialized to zero.
486    /// - `sum`: The sum of all data points, initialized to zero.
487    /// - `sum_sq`: The sum of squares of all data points, initialized to zero.
488    /// - `count`: The total number of data points, initialized to zero.
489    /// - `max_value`: The maximum allowable value for data points, set to the provided `max_value`.
490    ///
491    pub fn new_with_max(max_value: u64) -> Self {
492        LiveStatistics {
493            buckets: [0; BUCKET_COUNT],
494            min_val: u64::MAX,
495            max_val: 0,
496            sum: 0,
497            sum_sq: 0,
498            count: 0,
499            max_value,
500        }
501    }
502
503    #[inline]
504    fn value_to_bucket(&self, value: u64) -> usize {
505        if value >= self.max_value {
506            BUCKET_COUNT - 1
507        } else {
508            ((value as u128 * BUCKET_COUNT as u128) / self.max_value as u128) as usize
509        }
510    }
511
512    #[inline]
513    pub fn min(&self) -> u64 {
514        if self.count == 0 { 0 } else { self.min_val }
515    }
516
517    #[inline]
518    pub fn max(&self) -> u64 {
519        self.max_val
520    }
521
522    #[inline]
523    pub fn mean(&self) -> f64 {
524        if self.count == 0 {
525            0.0
526        } else {
527            self.sum as f64 / self.count as f64
528        }
529    }
530
531    #[inline]
532    pub fn stdev(&self) -> f64 {
533        if self.count == 0 {
534            return 0.0;
535        }
536        let mean = self.mean();
537        let variance = (self.sum_sq as f64 / self.count as f64) - (mean * mean);
538        if variance < 0.0 {
539            return 0.0;
540        }
541        #[cfg(feature = "std")]
542        return variance.sqrt();
543        #[cfg(not(feature = "std"))]
544        return sqrt(variance);
545    }
546
547    #[inline]
548    pub fn percentile(&self, percentile: f64) -> u64 {
549        if self.count == 0 {
550            return 0;
551        }
552
553        let target_count = (self.count as f64 * percentile) as u64;
554        let mut accumulated = 0u64;
555
556        for (bucket_idx, &bucket_count) in self.buckets.iter().enumerate() {
557            accumulated += bucket_count;
558            if accumulated >= target_count {
559                // Linear interpolation within the bucket
560                let bucket_start = (bucket_idx as u64 * self.max_value) / BUCKET_COUNT as u64;
561                let bucket_end = ((bucket_idx + 1) as u64 * self.max_value) / BUCKET_COUNT as u64;
562                let bucket_fraction = if bucket_count > 0 {
563                    (target_count - (accumulated - bucket_count)) as f64 / bucket_count as f64
564                } else {
565                    0.5
566                };
567                return bucket_start
568                    + ((bucket_end - bucket_start) as f64 * bucket_fraction) as u64;
569            }
570        }
571
572        self.max_val
573    }
574
575    /// Adds a value to the statistics.
576    #[inline]
577    pub fn record(&mut self, value: u64) {
578        if value < self.min_val {
579            self.min_val = value;
580        }
581        if value > self.max_val {
582            self.max_val = value;
583        }
584        self.sum += value;
585        self.sum_sq += value * value;
586        self.count += 1;
587
588        let bucket = self.value_to_bucket(value);
589        self.buckets[bucket] += 1;
590    }
591
592    #[inline]
593    pub fn len(&self) -> u64 {
594        self.count
595    }
596
597    #[inline]
598    pub fn is_empty(&self) -> bool {
599        self.count == 0
600    }
601
602    #[inline]
603    pub fn reset(&mut self) {
604        self.buckets.fill(0);
605        self.min_val = u64::MAX;
606        self.max_val = 0;
607        self.sum = 0;
608        self.sum_sq = 0;
609        self.count = 0;
610    }
611}

613/// A Specialized statistics object for CuDuration.
614/// It will also keep track of the jitter between the values.
615#[derive(Debug, Clone)]
616pub struct CuDurationStatistics {
617    bare: LiveStatistics,
618    jitter: LiveStatistics,
619    last_value: CuDuration,
620}

622impl CuDurationStatistics {
623    pub fn new(max: CuDuration) -> Self {
624        let CuDuration(max) = max;
625        CuDurationStatistics {
626            bare: LiveStatistics::new_with_max(max),
627            jitter: LiveStatistics::new_with_max(max),
628            last_value: CuDuration::default(),
629        }
630    }
631
632    #[inline]
633    pub fn min(&self) -> CuDuration {
634        CuDuration(self.bare.min())
635    }
636
637    #[inline]
638    pub fn max(&self) -> CuDuration {
639        CuDuration(self.bare.max())
640    }
641
642    #[inline]
643    pub fn mean(&self) -> CuDuration {
644        CuDuration(self.bare.mean() as u64) // CuDuration is in ns, it is ok.
645    }
646
647    #[inline]
648    pub fn percentile(&self, percentile: f64) -> CuDuration {
649        CuDuration(self.bare.percentile(percentile))
650    }

652    #[inline]
653    pub fn stddev(&self) -> CuDuration {
654        CuDuration(self.bare.stdev() as u64)

659        self.bare.len()
660    }

665    }
666
667    #[inline]
668    pub fn jitter_min(&self) -> CuDuration {
669        CuDuration(self.jitter.min())
670    }
671
672    #[inline]
673    pub fn jitter_max(&self) -> CuDuration {
674        CuDuration(self.jitter.max())
675    }
676
677    #[inline]
678    pub fn jitter_mean(&self) -> CuDuration {
679        CuDuration(self.jitter.mean() as u64)
680    }
681
682    #[inline]
683    pub fn jitter_stddev(&self) -> CuDuration {
684        CuDuration(self.jitter.stdev() as u64)
685    }
686
687    #[inline]
688    pub fn jitter_percentile(&self, percentile: f64) -> CuDuration {
689        CuDuration(self.jitter.percentile(percentile))
690    }
691
692    #[inline]
693    pub fn record(&mut self, value: CuDuration) {
694        let CuDuration(nanos) = value;
695        if self.bare.is_empty() {
696            self.bare.record(nanos);
697            self.last_value = value;
698            return;
699        }
700        self.bare.record(nanos);
701        let CuDuration(last_nanos) = self.last_value;
702        self.jitter.record(nanos.abs_diff(last_nanos));
703        self.last_value = value;
704    }
705
706    #[inline]
707    pub fn reset(&mut self) {
708        self.bare.reset();
709        self.jitter.reset();
710    }
