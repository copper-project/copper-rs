43/// Compile-time subsystem identity embedded in generated Copper applications.
44#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
45pub struct Subsystem {
46    id: Option<&'static str>,
47    code: u16,
48}
49
50impl Subsystem {
51    #[inline]
52    pub const fn new(id: Option<&'static str>, code: u16) -> Self {
53        Self { id, code }
54    }
55
56    #[inline]
57    pub const fn id(self) -> Option<&'static str> {
58        self.id
59    }
60
61    #[inline]

64    }
65}

73/// A trait that defines the structure and behavior of a CuApplication.
74///
75/// CuApplication is the normal, running on robot version of an application and its runtime.
76///
77/// The `CuApplication` trait outlines the necessary functions required for managing an application lifecycle,
78/// including configuration management, initialization, task execution, and runtime control. It is meant to be
79/// implemented by types that represent specific applications, providing them with unified control and execution features.
80///
81/// This is the more generic version that allows you to specify a custom unified logger.
82pub trait CuApplication<S: SectionStorage, L: UnifiedLogWrite<S> + 'static> {
83    /// Returns the original configuration as a string, typically loaded from a RON file.
84    /// This configuration represents the default settings for the application before any overrides.
85    fn get_original_config() -> String;
86
87    /// Creates a new application.
88    ///
89    /// # Arguments
90    ///
91    /// * `clock` - A `RobotClock` instance to be used for time-related operations in the implementing struct.
92    /// * `unified_logger` - A thread-safe, shared reference to `UnifiedLoggerWrite`, enabling logging functionalities.
93    /// * `config_override` - An optional `CuConfig` instance that allows overriding the default configuration values.
94    ///   - If `Some`, the provided configuration will be used.
95    ///   - If `None`, the default configuration will be applied.
96    ///
97    /// # Returns
98    ///
99    /// A result containing either:
100    /// - An instantiated object of the implementing type (`Self`), or
101    /// - A `CuResult` error in case of failure during initialization.
102    ///
103    fn new(
104        clock: RobotClock,
105        unified_logger: Arc<Mutex<L>>,
106        #[cfg(feature = "std")] config_override: Option<CuConfig>, // No config override in no-std, the bundled config is always the config
107    ) -> CuResult<Self>
108    where
109        Self: Sized;
110
111    /// Starts all tasks managed by the application/runtime.
112    ///
113    /// # Returns
114    /// * `Ok(())` - If all tasks are started successfully.
115    /// * `Err(CuResult)` - If an error occurs while attempting to start one
116    ///   or more tasks.
117    fn start_all_tasks(&mut self) -> CuResult<()>;
118
119    /// Executes a single iteration of copper-generated runtime (generating and logging one copperlist)
120    ///
121    /// # Returns
122    ///
123    /// * `CuResult<()>` - Returns `Ok(())` if the iteration completes successfully, or an error
124    ///   wrapped in `CuResult` if something goes wrong during execution.
125    ///
126    fn run_one_iteration(&mut self) -> CuResult<()>;
127
128    /// Runs indefinitely looping over run_one_iteration
129    ///
130    /// # Returns
131    ///
132    /// Returns a `CuResult<()>`, which indicates the success or failure of the
133    /// operation.
134    /// - On success, the result is `Ok(())`.
135    /// - On failure, an appropriate error wrapped in `CuResult` is returned.
136    fn run(&mut self) -> CuResult<()>;
137
138    /// Stops all tasks managed by the application/runtime.
139    ///
140    /// # Returns
141    ///
142    /// Returns a `CuResult<()>`, which indicates the success or failure of the
143    /// operation.
144    /// - On success, the result is `Ok(())`.
145    /// - On failure, an appropriate error wrapped in `CuResult` is returned.
146    ///
147    fn stop_all_tasks(&mut self) -> CuResult<()>;
148
149    /// Restore all tasks from the given frozen state
150    fn restore_keyframe(&mut self, freezer: &KeyFrame) -> CuResult<()>;
151}

153/// A trait that defines the structure and behavior of a simulation-enabled CuApplication.
154///
155/// CuSimApplication is the simulation version of an application and its runtime, allowing
156/// overriding of steps with simulated behavior.
157///
158/// The `CuSimApplication` trait outlines the necessary functions required for managing an application lifecycle
159/// in simulation mode, including configuration management, initialization, task execution, and runtime control.
160#[cfg(feature = "std")]
161pub trait CuSimApplication<S: SectionStorage, L: UnifiedLogWrite<S> + 'static> {
162    /// The type representing a simulation step that can be overridden
163    type Step<'z>;
164
165    /// Returns the original configuration as a string, typically loaded from a RON file.
166    /// This configuration represents the default settings for the application before any overrides.
167    fn get_original_config() -> String;
168
169    /// Creates a new simulation-enabled application.
170    ///
171    /// # Arguments
172    ///
173    /// * `clock` - A `RobotClock` instance to be used for time-related operations in the implementing struct.
174    /// * `unified_logger` - A thread-safe, shared reference to `UnifiedLoggerWrite`, enabling logging functionalities.
175    /// * `config_override` - An optional `CuConfig` instance that allows overriding the default configuration values.
176    ///   - If `Some`, the provided configuration will be used.
177    ///   - If `None`, the default configuration will be applied.
178    /// * `sim_callback` - A mutable function reference that allows overriding individual simulation steps.
179    ///   The callback receives a Step parameter and returns a SimOverride indicating how to handle the step.
180    ///
181    /// # Returns
182    ///
183    /// A result containing either:
184    /// - An instantiated object of the implementing type (`Self`), or
185    /// - A `CuResult` error in case of failure during initialization.
186    fn new(
187        clock: RobotClock,
188        unified_logger: Arc<Mutex<L>>,
189        config_override: Option<CuConfig>,
190        sim_callback: &mut impl for<'z> FnMut(Self::Step<'z>) -> SimOverride,
191    ) -> CuResult<Self>
192    where
193        Self: Sized;
194
195    /// Starts all tasks managed by the application/runtime in simulation mode.
196    ///
197    /// # Arguments
198    /// * `sim_callback` - A mutable function reference that allows overriding individual simulation steps.
199    ///
200    /// # Returns
201    /// * `Ok(())` - If all tasks are started successfully.
202    /// * `Err(CuResult)` - If an error occurs while attempting to start one
203    ///   or more tasks.
204    fn start_all_tasks(
205        &mut self,
206        sim_callback: &mut impl for<'z> FnMut(Self::Step<'z>) -> SimOverride,
207    ) -> CuResult<()>;
208
209    /// Executes a single iteration of copper-generated runtime in simulation mode.
210    ///
211    /// # Arguments
212    /// * `sim_callback` - A mutable function reference that allows overriding individual simulation steps.
213    ///
214    /// # Returns
215    ///
216    /// * `CuResult<()>` - Returns `Ok(())` if the iteration completes successfully, or an error
217    ///   wrapped in `CuResult` if something goes wrong during execution.
218    fn run_one_iteration(
219        &mut self,
220        sim_callback: &mut impl for<'z> FnMut(Self::Step<'z>) -> SimOverride,
221    ) -> CuResult<()>;
222
223    /// Runs indefinitely looping over run_one_iteration in simulation mode
224    ///
225    /// # Arguments
226    /// * `sim_callback` - A mutable function reference that allows overriding individual simulation steps.
227    ///
228    /// # Returns
229    ///
230    /// Returns a `CuResult<()>`, which indicates the success or failure of the
231    /// operation.
232    /// - On success, the result is `Ok(())`.
233    /// - On failure, an appropriate error wrapped in `CuResult` is returned.
234    fn run(
235        &mut self,
236        sim_callback: &mut impl for<'z> FnMut(Self::Step<'z>) -> SimOverride,
237    ) -> CuResult<()>;
238
239    /// Stops all tasks managed by the application/runtime in simulation mode.
240    ///
241    /// # Arguments
242    /// * `sim_callback` - A mutable function reference that allows overriding individual simulation steps.
243    ///
244    /// # Returns
245    ///
246    /// Returns a `CuResult<()>`, which indicates the success or failure of the
247    /// operation.
248    /// - On success, the result is `Ok(())`.
249    /// - On failure, an appropriate error wrapped in `CuResult` is returned.
250    fn stop_all_tasks(
251        &mut self,
252        sim_callback: &mut impl for<'z> FnMut(Self::Step<'z>) -> SimOverride,
253    ) -> CuResult<()>;
254
255    /// Restore all tasks from the given frozen state
256    fn restore_keyframe(&mut self, freezer: &KeyFrame) -> CuResult<()>;
257}

259/// Simulation-enabled applications that can replay a recorded CopperList verbatim.
260///
261/// This is the exact-output replay primitive used by deterministic re-sim flows:
262/// task outputs and bridge receives are overridden from the recorded CopperList,
263/// bridge sends are skipped, and an optional recorded keyframe can be injected
264/// verbatim when the current CL is expected to capture one.
265#[cfg(feature = "std")]
266pub trait CuRecordedReplayApplication<S: SectionStorage, L: UnifiedLogWrite<S> + 'static>:
267    CuSimApplication<S, L>
268{
269    /// The generated recorded CopperList payload set for this application.
270    type RecordedDataSet: CopperListTuple;
271
272    /// Replay one recorded CopperList exactly as logged.
273    fn replay_recorded_copperlist(
274        &mut self,
275        clock_mock: &RobotClockMock,
276        copperlist: &CopperList<Self::RecordedDataSet>,
277        keyframe: Option<&KeyFrame>,
278    ) -> CuResult<()>;
279}

281/// Simulation-enabled applications that can be instantiated for distributed replay.
282///
283/// This extends exact-output replay with the one extra capability the
284/// distributed engine needs: build a replayable app for a specific
285/// deployment `instance_id` while keeping app construction type-safe.
286#[cfg(feature = "std")]
287pub trait CuDistributedReplayApplication<S: SectionStorage, L: UnifiedLogWrite<S> + 'static>:
288    CuRecordedReplayApplication<S, L> + CuSubsystemMetadata
289{
290    /// Build this app for deterministic distributed replay.
291    fn build_distributed_replay(
292        clock: RobotClock,
293        unified_logger: Arc<Mutex<L>>,
294        instance_id: u32,
295        config_override: Option<CuConfig>,
296    ) -> CuResult<Self>
297    where
298        Self: Sized;
