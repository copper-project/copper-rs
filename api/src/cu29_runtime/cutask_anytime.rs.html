1//! Trait and types to implement an anytime Copper task.
2//!
3//! An anytime task splits its work into a mandatory minimum ([`CuAnytimeTask::base`])
4//! plus optional bounded improvements ([`CuAnytimeTask::refine`]). The task *reports*
5//! what each quantum achieved through [`AnytimeStatus`]; the runtime *decides* whether
6//! to schedule another quantum from that status stream and its configured time budget
7//! and quality target. The task never sees the policy, so implementations stay
8//! reusable under any policy.
9
10use crate::config::ComponentConfig;
11use crate::context::CuContext;
12use crate::cutask::{CuMsgPack, CuMsgPayload, Freezable};
13use crate::reflect::Reflect;
14use cu29_traits::CuResult;
15use cu29_units::si::f32::Ratio;
16
17/// Normalized quality of a published result: a dimensionless [`Ratio`] in
18/// `0.0..=1.0`, higher is better and `1.0` means no further improvement is
19/// meaningful. Sharing one scale across tasks keeps a configured quality target
20/// portable.
21pub type Quality = Ratio;
22
23/// Returned by [`CuAnytimeTask::base`] and [`CuAnytimeTask::refine`]; drives the
24/// runtime's refinement scheduling.
25///
26/// `Q` is [`CuAnytimeTask::Quality`]: [`Quality`] for tasks that can score their
27/// result, `()` for tasks that cannot.
28///
29/// After any `Ok` return, the output must be valid and hold the best result produced
30/// so far for the current job: a quantum that regresses or plateaus keeps its
31/// candidate in task-local state and leaves the output untouched. The runtime never
32/// buffers or rolls back the output, so it can publish it at any stop point, and the
33/// published quality is monotone even when the algorithm internally is not.
34#[derive(Debug, Clone, Copy, PartialEq)]
35pub enum AnytimeStatus<Q> {
36    /// The output holds the best result so far; further refinement may help.
37    Improved(Q),
38    /// Proactive yield: no further improvement is possible for this job. The output
39    /// holds the final result.
40    Converged(Q),
41    /// Proactive give-up: the algorithm diverged or reached an unrecoverable state
42    /// for this job. Refinement stops; the output is published as-is, so a task that
43    /// can no longer vouch even for its base result must clear the payload before
44    /// returning this. The next copperlist starts a fresh job.
45    Aborted,
46}
47
48/// A task producing a valid result from the bare-minimum compute, then improving it
49/// in bounded quanta for as long as the runtime allows.
50///
51/// Per job: `preprocess` → `base` → N × `refine` → `postprocess`, where N is chosen
52/// by the runtime (possibly 0: a time budget may suppress every refinement, but
53/// never the base computation).
54pub trait CuAnytimeTask: Freezable + Reflect {
55    type Input<'m>: CuMsgPack;
56    type Output<'m>: CuMsgPayload;
57    /// Resources required by the task.
58    type Resources<'r>;
59    /// Measure reported through [`AnytimeStatus`]: [`Quality`] for tasks that can
60    /// score their result, `()` for tasks that cannot. A quality target can only be
61    /// configured for tasks whose `Quality` is comparable to it, so a target on a
62    /// `()` task is rejected at compile time.
63    type Quality: Copy + PartialOrd;
64
65    /// Here you need to initialize everything your task will need for the duration
66    /// of its lifetime. The config allows you to access the configuration of the task.
67    fn new(_config: Option<&ComponentConfig>, _resources: Self::Resources<'_>) -> CuResult<Self>
68    where
69        Self: Sized;
70
71    /// Start is called between the creation of the task and the first call to
72    /// pre/base.
73    fn start(&mut self, _ctx: &CuContext) -> CuResult<()> {
74        Ok(())
75    }
76
77    /// This is a method called by the runtime before "base". This is a kind of best
78    /// effort, as soon as possible call to give a chance for the task to do some work
79    /// before to prepare to make "base" as short as possible.
80    fn preprocess(&mut self, _ctx: &CuContext) -> CuResult<()> {
81        Ok(())
82    }
83
84    /// Starts a new job and writes its minimum valid result into `output`.
85    ///
86    /// On `Ok`: `output` is valid and safe to publish, refinement state from the
87    /// preceding job has been reset, and later `refine()` calls improve this job.
88    /// The task must capture into its own per-job state everything refinement will
89    /// need from `input`: `refine()` does not receive the input (in background
90    /// placements refinement outlives the copperlist that carried it), and the task
91    /// knows the cheapest representation to retain.
92    ///
93    /// On `Err`: the job produced no valid output and the error propagates like a
94    /// `CuTask::process` failure.
95    fn base<'i, 'o>(
96        &mut self,
97        ctx: &CuContext,
98        input: &Self::Input<'i>,
99        output: &mut Self::Output<'o>,
100    ) -> CuResult<AnytimeStatus<Self::Quality>>;
101
102    /// Performs exactly one bounded refinement quantum.
103    ///
104    /// On `Ok` (any status), `output` is valid and holds the best result produced so
105    /// far for this job; see [`AnytimeStatus`] for the commit-only-improvements
106    /// contract.
107    ///
108    /// Anything a quantum could want to know about its own job the task already has:
109    /// it can count its quanta, read the clock through `ctx`, and remembers the last
110    /// quality it reported.
111    ///
112    /// This method must not contain an unbounded refinement loop: the runtime can
113    /// only observe time and quality *between* calls, so one call must be one
114    /// bounded quantum.
115    fn refine<'o>(
116        &mut self,
117        ctx: &CuContext,
118        output: &mut Self::Output<'o>,
119    ) -> CuResult<AnytimeStatus<Self::Quality>>;
120
121    /// This is a method called by the runtime after the job's refinement window has
122    /// closed. It is best effort a chance for the task to update some state out of
123    /// the critical path, for example to release scratch memory or maintain
124    /// statistics that are not time-critical for the robot.
125    fn postprocess(&mut self, _ctx: &CuContext) -> CuResult<()> {
126        Ok(())
127    }
128
129    /// Called to stop the task. It signals that `base`/`refine` won't be called
130    /// until start is called again.
131    fn stop(&mut self, _ctx: &CuContext) -> CuResult<()> {
132        Ok(())
133    }
134}

136#[cfg(test)]
137mod tests {
138    use super::*;
139    use crate::cutask::CuMsg;
140    use crate::input_msg;
141    use crate::output_msg;
142    use cu29_units::si::ratio::ratio;
143
144    /// Sums its input one increment per quantum: base publishes 0, each refine
145    /// commits one more increment until the captured input is fully consumed.
146    #[derive(Reflect)]
147    struct IncrementalSum {
148        target: u32,
149        acc: u32,
150    }
151
152    impl Freezable for IncrementalSum {}
153
154    impl CuAnytimeTask for IncrementalSum {
155        type Input<'m> = input_msg!(u32);
156        type Output<'m> = output_msg!(u32);
157        type Resources<'r> = ();
158        type Quality = Quality;
159
160        fn new(
161            _config: Option<&ComponentConfig>,
162            _resources: Self::Resources<'_>,
163        ) -> CuResult<Self> {
164            Ok(Self { target: 0, acc: 0 })
165        }
166
167        fn base<'i, 'o>(
168            &mut self,
169            _ctx: &CuContext,
170            input: &Self::Input<'i>,
171            output: &mut Self::Output<'o>,
172        ) -> CuResult<AnytimeStatus<Quality>> {
173            self.target = *input.payload().ok_or("no input")?;
174            self.acc = 0;
175            output.set_payload(self.acc);
176            Ok(AnytimeStatus::Improved(Quality::new::<ratio>(0.0)))
177        }
178
179        fn refine<'o>(
180            &mut self,
181            _ctx: &CuContext,
182            output: &mut Self::Output<'o>,
183        ) -> CuResult<AnytimeStatus<Quality>> {
184            if self.acc == self.target {
185                return Ok(AnytimeStatus::Converged(Quality::new::<ratio>(1.0)));
186            }
187            self.acc += 1;
188            output.set_payload(self.acc);
189            Ok(AnytimeStatus::Improved(Quality::new::<ratio>(
190                self.acc as f32 / self.target as f32,
191            )))
192        }
193    }
194
195    #[test]
196    fn base_then_refine_until_converged() {
197        let ctx = CuContext::new_with_clock();
198        let mut task = IncrementalSum::new(None, ()).unwrap();
199        let input = CuMsg::new(Some(3u32));
200        let mut output = CuMsg::new(None);
201
202        task.start(&ctx).unwrap();
203        task.preprocess(&ctx).unwrap();
204        let status = task.base(&ctx, &input, &mut output).unwrap();
205        assert!(matches!(status, AnytimeStatus::Improved(_)));
206        assert_eq!(output.payload(), Some(&0));
207
208        let mut best_quality = Quality::new::<ratio>(0.0);
209        for _ in 0..8 {
210            match task.refine(&ctx, &mut output).unwrap() {
211                AnytimeStatus::Improved(quality) => best_quality = quality,
212                AnytimeStatus::Converged(quality) => {
213                    best_quality = quality;
214                    break;
215                }
216                status => panic!("unexpected status: {status:?}"),
217            }
218        }
219        assert_eq!(output.payload(), Some(&3));
220        assert_eq!(best_quality.get::<ratio>(), 1.0);
221        task.postprocess(&ctx).unwrap();
222        task.stop(&ctx).unwrap();
223    }
224}
