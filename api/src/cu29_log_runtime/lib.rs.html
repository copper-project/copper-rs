75impl LoggerRuntime {
76    /// destination is the binary stream in which we will log the structured log.
77    /// `extra_text_logger` is the logger that will log the text logs in real time. This is slow and only for debug builds.
78    pub fn init(
79        clock: RobotClock,
80        destination: impl WriteStream<CuLogEntry> + 'static,
81        #[allow(unused_variables)] extra_text_logger: Option<impl Log + 'static>,
82    ) -> Self {
83        let runtime = LoggerRuntime {};
84        STRUCTURED_LOG_BYTES.store(0, Ordering::Relaxed);
85
86        // If WRITER is already initialized, update the inner value.
87        // This should only be useful for unit testing.
88        if let Some((writer, _)) = WRITER.get() {
89            #[cfg(not(feature = "std"))]
90            let mut writer_guard = writer.lock();
91            #[cfg(feature = "std")]
92            let mut writer_guard = writer.lock().unwrap_or_else(|e| e.into_inner());
93            *writer_guard = Box::new(destination);
94        } else {
95            #[cfg(not(feature = "std"))]
96            WRITER.call_once(|| (Mutex::new(Box::new(destination)), clock));
97            #[cfg(feature = "std")]
98            WRITER
99                .set((Mutex::new(Box::new(destination)), clock))
100                .unwrap();
101        }
102        #[cfg(debug_assertions)]
103        #[cfg(feature = "std")]
104        if let Some(logger) = extra_text_logger {
105            let mut extra_text_logger = EXTRA_TEXT_LOGGER.write().unwrap();
106            *extra_text_logger = Some(Box::new(logger) as Box<dyn Log>);
107        }
108
109        runtime
110    }
111
112    pub fn flush(&self) {
113        // no op in no_std TODO(gbin): check if it will be needed in no_std at some point.
114        #[cfg(feature = "std")]
115        if let Some((writer, _clock)) = WRITER.get() {
116            if let Ok(mut writer) = writer.lock() {
117                if let Err(err) = writer.flush() {
118                    eprintln!("cu29_log: Failed to flush writer: {err}");
119                }
120            } else {
121                eprintln!("cu29_log: Failed to lock writer.");
122            }
123        } else {
124            eprintln!("cu29_log: Logger not initialized.");
125        }
126    }
127}

129impl Drop for LoggerRuntime {
130    fn drop(&mut self) {
131        self.flush();
132        // Assume on no-std that there is no buffering. TODO(gbin): check if this hold true.
133        #[cfg(feature = "std")]
134        if let Some((mutex, _clock)) = WRITER.get()
135            && let Ok(mut writer_guard) = mutex.lock()
136        {
137            // Replace the current WriteStream with a DummyWriteStream
138            *writer_guard = Box::new(DummyWriteStream);
139        }
140    }
141}

143/// Function called from generated code to log data.
144/// It moves entry by design, it will be absorbed in the queue.
145#[inline(always)]
146pub fn log(entry: &mut CuLogEntry) -> CuResult<()> {
147    let (writer, clock) = match WRITER.get() {
148        Some((writer, clock)) => (writer, clock),
149        None => return Err("Logger not initialized.".into()),
150    };
151    entry.time = clock.now();
152
153    #[cfg(not(feature = "std"))]
154    {
155        let mut guard = writer.lock();
156        guard.log(entry)?;
157        if let Some(bytes) = guard.last_log_bytes() {
158            STRUCTURED_LOG_BYTES.fetch_add(bytes, Ordering::Relaxed);
159        }
160    }
161
162    #[cfg(feature = "std")]
163    {
164        let mut guard = match writer.lock() {
165            Ok(guard) => guard,
166            Err(err) => {
167                eprintln!("cu29_log: Logger mutex poisoned, recovering.");
168                err.into_inner()
169            }
170        };
171        if let Err(err) = guard.log(entry) {
172            eprintln!("Failed to log data: {err}");
173        } else if let Some(bytes) = guard.last_log_bytes() {
174            STRUCTURED_LOG_BYTES.fetch_add(bytes, Ordering::Relaxed);
175        }
176    }
177    Ok(())
178}

185/// This version of log is only compiled in debug mode
186/// This allows a normal logging framework to be bridged.
187#[cfg(debug_assertions)]
188pub fn log_debug_mode(
189    entry: &mut CuLogEntry,
190    _format_str: &str, // this is the missing info at runtime.
191    _param_names: &[&str],
192) -> CuResult<()> {
193    log(entry)?;
194
195    // and the bridging is only available in std.
196    #[cfg(feature = "std")]
197    extra_log(entry, _format_str, _param_names)?;
198
199    Ok(())
200}

202#[cfg(debug_assertions)]
203#[cfg(feature = "std")]
204fn extra_log(entry: &mut CuLogEntry, format_str: &str, param_names: &[&str]) -> CuResult<()> {
205    let guarded_logger = EXTRA_TEXT_LOGGER.read().unwrap();
206    let Some(logger) = guarded_logger.as_ref() else {
207        return Ok(());
208    };
209
210    // transform the slice into a hashmap
211    let params: Vec<String> = entry.params.iter().map(|v| v.to_string()).collect();
212    let named_params: HashMap<String, String> = param_names
213        .iter()
214        .zip(params.iter())
215        .map(|(name, value)| (name.to_string(), value.clone()))
216        .collect();
217
218    // Convert Copper log level to the standard log level
219    // Note: CuLogLevel::Critical is mapped to log::Level::Error because the `log` crate
220    // does not have a `Critical` level. `Error` is the highest severity level available
221    // in the `log` crate, making it the closest equivalent.
222    let log_level = match entry.level {
223        CuLogLevel::Debug => log::Level::Debug,
224        CuLogLevel::Info => log::Level::Info,
225        CuLogLevel::Warning => log::Level::Warn,
226        CuLogLevel::Error => log::Level::Error,
227        CuLogLevel::Critical => log::Level::Error,
228    };
229
230    let logline = format_logline(
231        entry.time,
232        entry.level,
233        format_str,
234        params.as_slice(),
235        &named_params,
236    )?;
237    logger.log(
238        &log::Record::builder()
239            .args(format_args!("{logline}"))
240            .level(log_level)
241            .target("cu29_log")
242            .module_path_static(Some("cu29_log"))
243            .file_static(Some("cu29_log"))
244            .line(Some(0))
245            .build(),
246    );
247    Ok(())
248}

251#[cfg(feature = "std")]
252pub struct OwningIoWriter<W: Write> {
253    writer: BufWriter<W>,
254    bytes_written: usize,
255}
256
257#[cfg(feature = "std")]
258impl<W: Write> OwningIoWriter<W> {
259    pub fn new(writer: W) -> Self {
260        Self {
261            writer: BufWriter::new(writer),
262            bytes_written: 0,
263        }
264    }
265
266    pub fn bytes_written(&self) -> usize {
267        self.bytes_written
268    }
269
270    pub fn flush(&mut self) -> Result<(), EncodeError> {
271        self.writer.flush().map_err(|inner| EncodeError::Io {
272            inner,
273            index: self.bytes_written,
274        })
275    }
276}
277
278#[cfg(feature = "std")]
279impl<W: Write> Writer for OwningIoWriter<W> {
280    #[inline(always)]
281    fn write(&mut self, bytes: &[u8]) -> Result<(), EncodeError> {
282        self.writer
283            .write_all(bytes)
284            .map_err(|inner| EncodeError::Io {
285                inner,
286                index: self.bytes_written,
287            })?;
288        self.bytes_written += bytes.len();
289        Ok(())
290    }
291}

293/// This allows this crate to be used outside of Copper (ie. decoupling it from the unifiedlog.
294#[cfg(feature = "std")]
295pub struct SimpleFileWriter {
296    path: PathBuf,
297    encoder: EncoderImpl<OwningIoWriter<File>, Configuration>,
298}
299
300#[cfg(feature = "std")]
301impl SimpleFileWriter {
302    pub fn new(path: &PathBuf) -> CuResult<Self> {
303        let file = std::fs::OpenOptions::new()
304            .create(true)
305            .truncate(true)
306            .write(true)
307            .open(path)
308            .map_err(|e| format!("Failed to open file: {e:?}"))?;
309
310        let writer = OwningIoWriter::new(file);
311        let encoder = EncoderImpl::new(writer, bincode::config::standard());
312
313        Ok(SimpleFileWriter {
314            path: path.clone(),
315            encoder,
316        })
317    }
318}
319
320#[cfg(feature = "std")]
321impl Debug for SimpleFileWriter {
322    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
323        write!(f, "SimpleFileWriter for path {:?}", self.path)
324    }
325}
326
327#[cfg(feature = "std")]
328impl WriteStream<CuLogEntry> for SimpleFileWriter {
329    #[inline(always)]
330    fn log(&mut self, obj: &CuLogEntry) -> CuResult<()> {
331        obj.encode(&mut self.encoder)
332            .map_err(|e| format!("Failed to write to file: {e:?}"))?;
333        Ok(())
334    }
335
336    fn flush(&mut self) -> CuResult<()> {
337        self.encoder
338            .writer()
339            .flush()
340            .map_err(|e| format!("Failed to flush file: {e:?}"))?;
341        Ok(())
342    }
343}
