24/// This is the configuration of a component (like a task config or a monitoring config):w
25/// It is a map of key-value pairs.
26/// It is given to the new method of the task implementation.
27#[derive(Serialize, Deserialize, Debug, Clone, Default)]
28pub struct ComponentConfig(pub HashMap<String, Value>);
29
30impl Display for ComponentConfig {
31    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
32        let mut first = true;
33        let ComponentConfig(config) = self;
34        write!(f, "{{")?;
35        for (key, value) in config.iter() {
36            if !first {
37                write!(f, ", ")?;
38            }
39            write!(f, "{key}: {value}")?;
40            first = false;
41        }
42        write!(f, "}}")
43    }
44}

116macro_rules! impl_from_value_for_int {
117    ($($target:ty),* $(,)?) => {
118        $(
119            impl From<Value> for $target {
120                fn from(value: Value) -> Self {
121                    if let Value(RonValue::Number(num)) = value {
122                        match num {
123                            Number::I8(n) => n as $target,
124                            Number::I16(n) => n as $target,
125                            Number::I32(n) => n as $target,
126                            Number::I64(n) => n as $target,
127                            Number::U8(n) => n as $target,
128                            Number::U16(n) => n as $target,
129                            Number::U32(n) => n as $target,
130                            Number::U64(n) => n as $target,
131                            Number::F32(_) | Number::F64(_) => {
132                                panic!("Expected an integer Number variant but got {num:?}")
133                            }
134                        }
135                    } else {
136                        panic!("Expected a Number variant but got {value:?}")
137                    }
138                }
139            }
140        )*
141    };
142}

172impl Display for Value {
173    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
174        let Value(value) = self;
175        match value {
176            RonValue::Number(n) => {
177                let s = match n {
178                    Number::I8(n) => n.to_string(),
179                    Number::I16(n) => n.to_string(),
180                    Number::I32(n) => n.to_string(),
181                    Number::I64(n) => n.to_string(),
182                    Number::U8(n) => n.to_string(),
183                    Number::U16(n) => n.to_string(),
184                    Number::U32(n) => n.to_string(),
185                    Number::U64(n) => n.to_string(),
186                    Number::F32(n) => n.0.to_string(),
187                    Number::F64(n) => n.0.to_string(),
188                    _ => panic!("Expected a Number variant but got {value:?}"),
189                };
190                write!(f, "{s}")
191            }
192            RonValue::String(s) => write!(f, "{s}"),
193            RonValue::Bool(b) => write!(f, "{b}"),
194            RonValue::Map(m) => write!(f, "{m:?}"),
195            RonValue::Char(c) => write!(f, "{c:?}"),
196            RonValue::Unit => write!(f, "unit"),
197            RonValue::Option(o) => write!(f, "{o:?}"),
198            RonValue::Seq(s) => write!(f, "{s:?}"),
199            RonValue::Bytes(bytes) => write!(f, "{bytes:?}"),
200        }
201    }
202}

264/// This represents a connection between 2 tasks (nodes) in the configuration graph.
265#[derive(Serialize, Deserialize, Debug, Clone)]
266pub struct Cnx {
267    /// Source node id.
268    src: String,
269
270    // Destination node id.
271    dst: String,
272
273    /// Message type exchanged between src and dst.
274    pub msg: String,
275
276    /// Tells Copper to batch messages before sending the buffer to the next node.
277    /// If None, Copper will just send 1 message at a time.
278    /// If Some(n), Copper will batch n messages before sending the buffer.
279    pub batch: Option<u32>,
280
281    /// Tells Copper if it needs to log the messages.
282    pub store: Option<bool>,

329/// The config is a list of tasks and their connections.
330#[derive(Serialize, Deserialize, Default)]
331struct CuConfigRepresentation {
332    tasks: Vec<Node>,
333    cnx: Vec<Cnx>,
334    monitor: Option<MonitorConfig>,
335    logging: Option<LoggingConfig>,
336}
337
338impl<'de> Deserialize<'de> for CuConfig {
339    /// This is a custom serialization to make this implementation independent of petgraph.
340    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
341    where
342        D: Deserializer<'de>,
343    {
344        let representation =
345            CuConfigRepresentation::deserialize(deserializer).map_err(serde::de::Error::custom)?;
346
347        let mut cuconfig = CuConfig::default();
348        for task in representation.tasks {
349            cuconfig.add_node(task);
350        }
351
352        for c in representation.cnx {
353            let src = cuconfig
354                .graph
355                .node_indices()
356                .find(|i| cuconfig.graph[*i].id == c.src)
357                .expect("Source node not found");
358            let dst = cuconfig
359                .graph
360                .node_indices()
361                .find(|i| cuconfig.graph[*i].id == c.dst)
362                .unwrap_or_else(|| panic!("Destination {} node not found", c.dst));
363            cuconfig.connect_ext(
364                src.index() as NodeId,
365                dst.index() as NodeId,
366                &c.msg,
367                c.batch,
368                c.store,
369            );
370        }
371        cuconfig.monitor = representation.monitor;
372        cuconfig.logging = representation.logging;
373        Ok(cuconfig)
374    }
375}

377impl Serialize for CuConfig {
378    /// This is a custom serialization to make this implementation independent of petgraph.
379    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
380    where
381        S: Serializer,
382    {
383        let tasks: Vec<Node> = self
384            .graph
385            .node_indices()
386            .map(|idx| self.graph[idx].clone())
387            .collect();
388
389        let cnx: Vec<Cnx> = self
390            .graph
391            .edge_indices()
392            .map(|edge| self.graph[edge].clone())
393            .collect();
394
395        CuConfigRepresentation {
396            tasks,
397            cnx,
398            monitor: self.monitor.clone(),
399            logging: self.logging.clone(),
400        }
401        .serialize(serializer)
402    }
403}

423    /// Get the node with the given id.

425    pub fn get_node(&self, node_id: NodeId) -> Option<&Node> {
426        self.graph.node_weight(node_id.into())
427    }
428
429    /// Get the node with the given id mutably.
430    #[allow(dead_code)] // Used in proc macro
431    pub fn get_node_mut(&mut self, node_id: NodeId) -> Option<&mut Node> {
432        self.graph.node_weight_mut(node_id.into())
433    }
434
435    /// this is more like infer from the connections of this node.
436    #[allow(dead_code)] // Used in proc macro
437    pub fn get_node_output_msg_type(&self, node_id: &str) -> Option<String> {
438        self.graph.node_indices().find_map(|node_index| {
439            if let Some(node) = self.get_node(node_index.index() as u32) {
440                if node.id != node_id {
441                    return None;
442                }
443                let edges = self.get_src_edges(node_index.index() as u32);
444                if edges.is_empty() {
445                    panic!("A CuSrcTask is configured with no task connected to it.")
446                }
447                let cnx = self
448                    .graph
449                    .edge_weight(EdgeIndex::new(edges[0]))
450                    .expect("Found an cnx id but could not retrieve it back");
451                return Some(cnx.msg.clone());
452            }
453            None
454        })
455    }
456
457    /// this is more like infer from the connections of this node.
458    #[allow(dead_code)] // Used in proc macro
459    pub fn get_node_input_msg_type(&self, node_id: &str) -> Option<String> {
460        self.graph.node_indices().find_map(|node_index| {
461            if let Some(node) = self.get_node(node_index.index() as u32) {
462                if node.id != node_id {
463                    return None;
464                }
465                let edges = self.get_dst_edges(node_index.index() as u32);
466                if edges.is_empty() {
467                    panic!("A CuSinkTask is configured with no task connected to it.")
468                }
469                let cnx = self
470                    .graph
471                    .edge_weight(EdgeIndex::new(edges[0]))
472                    .expect("Found an cnx id but could not retrieve it back");
473                return Some(cnx.msg.clone());
474            }
475            None
476        })
477    }

479    /// Get the list of edges that are connected to the given node as a source.

508    /// Adds an edge between two nodes/tasks in the configuration graph.
509    /// msg_type is the type of message exchanged between the two nodes/tasks.
510    /// batch is the number of messages to batch before sending the buffer.
511    /// store tells Copper if it needs to log the messages.
512    pub fn connect_ext(
513        &mut self,
514        source: NodeId,
515        target: NodeId,
516        msg_type: &str,
517        batch: Option<u32>,
518        store: Option<bool>,
519    ) {
520        self.graph.add_edge(
521            source.into(),
522            target.into(),
523            Cnx {
524                src: self
525                    .get_node(source)
526                    .expect("Source node not found")
527                    .id
528                    .clone(),
529                dst: self
530                    .get_node(target)
531                    .expect("Target node not found")
532                    .id
533                    .clone(),
534                msg: msg_type.to_string(),
535                batch,
536                store,
537            },
538        );
539    }

572    /// Render the configuration graph in the dot format.
573    pub fn render(&self, output: &mut dyn std::io::Write) {
574        writeln!(output, "digraph G {{").unwrap();
575
576        for index in self.graph.node_indices() {
577            let node = &self.graph[index];
578            let config_str = match &node.config {
579                Some(config) => {
580                    let config_str = config
581                        .0
582                        .iter()
583                        .map(|(k, v)| format!("<B>{k}</B> = {v}<BR ALIGN=\"LEFT\"/>"))
584                        .collect::<Vec<String>>()
585                        .join("\n");
586                    format!("____________<BR/><BR ALIGN=\"LEFT\"/>{config_str}")
587                }
588                None => String::new(),
589            };
590            writeln!(output, "{} [", index.index()).unwrap();
591            writeln!(output, "shape=box,").unwrap();
592            writeln!(output, "style=\"rounded, filled\",").unwrap();
593            writeln!(output, "fontname=\"Noto Sans\"").unwrap();
594
595            let is_src = self.get_dst_edges(index.index() as NodeId).is_empty();
596            let is_sink = self.get_src_edges(index.index() as NodeId).is_empty();
597            if is_src {
598                writeln!(output, "fillcolor=lightgreen,").unwrap();
599            } else if is_sink {
600                writeln!(output, "fillcolor=lightblue,").unwrap();
601            } else {
602                writeln!(output, "fillcolor=lightgrey,").unwrap();
603            }
604            writeln!(output, "color=grey,").unwrap();
605
606            writeln!(output, "labeljust=l,").unwrap();
607            writeln!(
608                output,
609                "label=< <FONT COLOR=\"red\"><B>{}</B></FONT> <FONT COLOR=\"dimgray\">[{}]</FONT><BR ALIGN=\"LEFT\"/>{} >",
610                node.id,
611                node.get_type(),
612                config_str
613            )
614                .unwrap();
615
616            writeln!(output, "];").unwrap();
617        }
618        for edge in self.graph.edge_indices() {
619            let (src, dst) = self.graph.edge_endpoints(edge).unwrap();
620
621            let cnx = &self.graph[edge];
622            let msg = encode_text(&cnx.msg);
623            writeln!(
624                output,
625                "{} -> {} [label=< <B><FONT COLOR=\"gray\">{}</FONT></B> >];",
626                src.index(),
627                dst.index(),
628                msg
629            )
630            .unwrap();
631        }
632        writeln!(output, "}}").unwrap();
633    }

658impl LoggingConfig {
659    /// Validate the logging configuration to ensure section pre-allocation sizes do not exceed slab sizes.
660    pub fn validate(&self) -> CuResult<()> {
661        if let Some(section_size_mib) = self.section_size_mib {
662            if let Some(slab_size_mib) = self.slab_size_mib {
663                if section_size_mib > slab_size_mib {
664                    return Err(CuError::from(format!("Section size ({} MiB) cannot be larger than slab size ({} MiB). Adjust the parameters accordingly.", section_size_mib, slab_size_mib)));
665                }
666            }
667        }
668
669        Ok(())
670    }

673/// Read a copper configuration from a file.
674pub fn read_configuration(config_filename: &str) -> CuResult<CuConfig> {
675    let config_content = read_to_string(config_filename).map_err(|e| {
676        CuError::from(format!(
677            "Failed to read configuration file: {:?}",
678            &config_filename
679        ))
680        .add_cause(e.to_string().as_str())
681    })?;
682    read_configuration_str(config_content)
683}

685/// Read a copper configuration from a String.
686pub fn read_configuration_str(config_content: String) -> CuResult<CuConfig> {
687    let cuconfig = CuConfig::deserialize_ron(&config_content);
688    cuconfig.validate_logging_config()?;
689

789    // this test makes sure the edge id is suitable to be used to sort the inputs of a task
790    #[test]
791    fn test_deserialization_edge_id_assignment() {
792        // note here that the src1 task is added before src2 in the tasks array,
793        // however, src1 connection is added AFTER src2 in the cnx array
794        let txt = r#"( 
795            tasks: [(id: "src1", type: "a"), (id: "src2", type: "b"), (id: "sink", type: "c")],
796            cnx: [(src: "src2", dst: "sink", msg: "msg1"), (src: "src1", dst: "sink", msg: "msg2")]
797        )"#;
798        let config = CuConfig::deserialize_ron(txt);
799        assert!(config.validate_logging_config().is_ok());
800
801        // the node id depends on the order in which the tasks are added
802        let src1_id = 0;
803        assert_eq!(config.get_node(src1_id).unwrap().id, "src1");
804        let src2_id = 1;
805        assert_eq!(config.get_node(src2_id).unwrap().id, "src2");
806
807        // the edge id depends on the order the connection is created
808        // the src2 was added second in the tasks, but the connection was added first
809        let src1_edge_id = *config.get_src_edges(src1_id).first().unwrap();
810        assert_eq!(src1_edge_id, 1);
811        let src2_edge_id = *config.get_src_edges(src2_id).first().unwrap();
812        assert_eq!(src2_edge_id, 0);
813    }
