13/// This macro is used to log a message with parameters.
14/// The first parameter is a string literal that represents the message to be logged.
15/// Only `{}` is supported as a placeholder for parameters.
16/// The rest of the parameters are the values to be logged.
17/// The parameters can be named or unnamed.
18/// Named parameters are specified as `name = value`.
19/// Unnamed parameters are specified as `value`.
20/// # Example
21/// ```ignore
22/// use cu29_log_derive::debug;
23/// let a = 1;
24/// let b = 2;
25/// debug!("a = {}, b = {}", my_value = a, b); // named and unnamed parameters
26/// ```
27///
28/// You can retrieve this data using the log_reader generated with your project and giving it the
29/// unified .copper log file and the string index file generated at compile time.
30///
31/// Note: In debug mode, the log will also be printed to the console. (ie slooow).
32/// In release mode, the log will be only be written to the unified logger.
33#[proc_macro]
34pub fn debug(input: TokenStream) -> TokenStream {
35    let parser = syn::punctuated::Punctuated::<Expr, Token![,]>::parse_terminated;
36    let exprs = parser.parse(input).expect("Failed to parse input");
37
38    let mut exprs_iter = exprs.iter();
39
40    let msg_expr = exprs_iter.next().expect("Expected at least one expression");
41    let (index, _msg) = if let Expr::Lit(ExprLit {
42        lit: Lit::Str(msg), ..
43    }) = msg_expr
44    {
45        let msg = msg.value();
46        let index = intern_string(&msg).expect("Failed to insert log string.");
47        (index, msg)
48    } else {
49        panic!("The first parameter of the argument needs to be a string literal.");
50    };
51    let prefix = quote! {
52        let mut log_entry = CuLogEntry::new(#index);
53    };
54
55    let mut unnamed_params = vec![];
56    let mut named_params = vec![];
57
58    for expr in exprs_iter {
59        if let Expr::Assign(ExprAssign { left, right, .. }) = expr {
60            named_params.push((left, right));
61        } else {
62            unnamed_params.push(expr);
63        }
64    }
65
66    let unnamed_prints = unnamed_params.iter().map(|value| {
67        quote! {
68            let param = to_value(#value).expect("Failed to convert a parameter to a Value");
69            log_entry.add_param(ANONYMOUS, param);
70        }
71    });
72
73    let named_prints = named_params.iter().map(|(name, value)| {
74        let index = intern_string(quote!(#name).to_string().as_str())
75            .expect("Failed to insert log string.");
76        quote! {
77            let param = to_value(#value).expect("Failed to convert a parameter to a Value");
78            log_entry.add_param(#index, param);
79        }
80    });
81
82    #[cfg(not(debug_assertions))]
83    let log_stmt = quote! {
84        let r = log(&mut log_entry);
85    };
86
87    #[cfg(debug_assertions)]
88    let log_stmt = {
89        let keys: Vec<_> = named_params
90            .iter()
91            .map(|(name, _)| {
92                let name_str = quote!(#name).to_string(); // Convert the expression to a token stream, then to a string
93                let lit_str = syn::LitStr::new(&name_str, name.span()); // Create a string literal with the original span
94                quote!(#lit_str)
95            })
96            .collect();
97        quote! {
98            let r = log_debug_mode(&mut log_entry, #_msg, &[#(#keys),*]);
99        }
100    };
101
102    let postfix = quote! {
103        #log_stmt
104        if let Err(e) = r {
105            eprintln!("Warning: Failed to log: {}", e);
106            let backtrace = std::backtrace::Backtrace::capture();
107            eprintln!("{:?}", backtrace);
108        }
109    };
110
111    let expanded = quote! {
112        {
113            #prefix
114            #(#unnamed_prints)*
115            #(#named_prints)*
116            #postfix
117        }
118    };
119
120    expanded.into()
121}
