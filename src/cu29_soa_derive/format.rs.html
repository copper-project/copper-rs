9/// A utility method to ease up the debugging of the macro generated code by formatting it with rustfmt.
10#[allow(dead_code)]
11pub(crate) fn rustfmt_generated_code(code: String) -> String {
12    let mut rustfmt = Command::new("rustfmt")
13        .arg("--emit")
14        .arg("stdout")
15        .stdin(std::process::Stdio::piped())
16        .stdout(std::process::Stdio::piped())
17        .spawn()
18        .expect("Failed to spawn rustfmt");
19
20    {
21        let stdin = rustfmt.stdin.as_mut().expect("Failed to open stdin");
22        stdin
23            .write_all(code.as_bytes())
24            .expect("Failed to write to stdin");
25    }
26
27    let output = rustfmt.wait_with_output().expect("Failed to read stdout");
28    String::from_utf8(output.stdout).expect("Output was not valid UTF-8")
29}

32fn create_black_theme() -> Theme {

37        b: 0,

43/// A utility method to ease up the debugging of the macro generated code by highlighting it.
44#[allow(dead_code)]
45pub(crate) fn highlight_rust_code(code: String) -> String {
46    let ps = SyntaxSet::load_defaults_newlines();
47    let syntax = ps.find_syntax_by_extension("rs").unwrap();
48    let theme = create_black_theme();
49    let mut h = HighlightLines::new(syntax, &theme);
50
51    let mut highlighted_code = String::new();
52
53    for line in LinesWithEndings::from(&code) {
54        let ranges: Vec<(Style, &str)> = h.highlight_line(line, &ps).unwrap();
55        let escaped = as_24_bit_terminal_escaped(&ranges[..], true);
56        highlighted_code.push_str(&escaped);
57    }
58
59    highlighted_code
60}
