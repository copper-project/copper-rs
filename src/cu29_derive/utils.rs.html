5/// Small tool to create a valid enum entry from an identifier.
6pub(crate) fn config_id_to_enum(id: &str) -> String {
7    let mut candidate = id
8        .chars()
9        .map(|c| if c.is_alphanumeric() { c } else { '_' })
10        .collect::<String>();
11
12    candidate = candidate.to_case(Case::Pascal);
13
14    if candidate.chars().next().is_some_and(|c| c.is_ascii_digit()) {
15        candidate.insert(0, '_');
16    }
17
18    candidate
19}

38pub fn caller_crate_root() -> PathBuf {
39    let crate_name =
40        std::env::var("CARGO_PKG_NAME").expect("failed to read ENV var `CARGO_PKG_NAME`!");
41    let current_dir = std::env::current_dir().expect("failed to unwrap env::current_dir()!");
42    let search_entry = format!("name=\"{crate_name}\"");
43    for entry in WalkDir::new(&current_dir)
44        .into_iter()
45        .filter_entry(|e| !e.file_name().eq_ignore_ascii_case("target"))
46    {
47        let Ok(entry) = entry else {
48            continue;
49        };
50        if !entry.file_type().is_file() {
51            continue;
52        }
53        let Some(file_name) = entry.path().file_name() else {
54            continue;
55        };
56        if !file_name.eq_ignore_ascii_case("Cargo.toml") {
57            continue;
58        }
59        let Ok(cargo_toml) = std::fs::read_to_string(entry.path()) else {
60            continue;
61        };
62        if cargo_toml
63            .chars()
64            .filter(|&c| !c.is_whitespace())
65            .collect::<String>()
66            .contains(search_entry.as_str())
67        {
68            return entry.path().parent().unwrap().to_path_buf();
69        }
70    }
71    current_dir
72}

78    fn is_valid_rust_identifier(input: &str) -> bool {
79        if input.is_empty() {
80            return false;
81        }
82
83        // Check if the first character is valid
84        let mut chars = input.chars();
85        if let Some(first) = chars.next() {
86            if !first.is_alphabetic() && first != '_' {
87                return false;
88            }
89        }
90
91        // Check the rest of the characters
92        if !chars.all(|c| c.is_alphanumeric() || c == '_') {
93            return false;
94        }
95
96        // Check if it's a Rust keyword (can use a set of known keywords)
97        let keywords = [
98            "as", "break", "const", "continue", "crate", "else", "enum", "extern", "false", "fn",
99            "for", "if", "impl", "in", "let", "loop", "match", "mod", "move", "mut", "pub", "ref",
100            "return", "self", "Self", "static", "struct", "super", "trait", "true", "type",
101            "unsafe", "use", "where", "while",
102        ];
103
104        !keywords.contains(&input)
105    }
