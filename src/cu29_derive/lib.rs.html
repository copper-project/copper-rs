43/// Generates the CopperList content type from a config.
44/// gen_cumsgs!("path/to/config.toml")
45/// It will create a new type called CuMsgs you can pass to the log reader for decoding:
46#[proc_macro]
47pub fn gen_cumsgs(config_path_lit: TokenStream) -> TokenStream {
48    let config = parse_macro_input!(config_path_lit as LitStr).value();
49    if !std::path::Path::new(&config_full_path(&config)).exists() {
50        return return_error(format!(
51            "The configuration file `{config}` does not exist. Please provide a valid path."
52        ));
53    }
54    #[cfg(feature = "macro_debug")]
55    eprintln!("[gen culist support with {config:?}]");
56    let cuconfig = match read_config(&config) {
57        Ok(cuconfig) => cuconfig,
58        Err(e) => return return_error(e.to_string()),
59    };
60    let runtime_plan: CuExecutionLoop = match compute_runtime_plan(&cuconfig) {
61        Ok(plan) => plan,
62        Err(e) => return return_error(format!("Could not compute runtime plan: {e}")),
63    };
64
65    // Give a name compatible with a struct to match the task ids to their output in the CuMsgs tuple.
66    let all_tasks_member_ids: Vec<String> = cuconfig
67        .get_all_nodes(None) // FIXME(gbin): Multimission
68        .iter()
69        .map(|(_, node)| utils::config_id_to_struct_member(node.get_id().as_str()))
70        .collect();
71
72    // All accesses are linear on the culist but the id of the tasks is random (determined by the Ron declaration order).
73    // This records the task ids in call order.
74    let taskid_order: Vec<usize> = runtime_plan
75        .steps
76        .iter()
77        .filter_map(|unit| match unit {
78            CuExecutionUnit::Step(step) => Some(step.node_id as usize),
79            _ => None,
80        })
81        .collect();
82
83    #[cfg(feature = "macro_debug")]
84    eprintln!(
85        "[The CuMsgs matching tasks ids are {:?}]",
86        taskid_order
87            .iter()
88            .map(|i| all_tasks_member_ids[*i].clone())
89            .collect::<Vec<_>>()
90    );
91
92    let support = gen_culist_support(&runtime_plan, &taskid_order, &all_tasks_member_ids);
93
94    let with_uses = quote! {
95        mod cumsgs {
96            use cu29::bincode::Encode;
97            use cu29::bincode::enc::Encoder;
98            use cu29::bincode::error::EncodeError;
99            use cu29::bincode::Decode;
100            use cu29::bincode::de::Decoder;
101            use cu29::bincode::error::DecodeError;
102            use cu29::copperlist::CopperList;
103            use cu29::cutask::CuMsgMetadata;
104            use cu29::cutask::CuMsg;
105            #support
106        }
107        use cumsgs::CuMsgs;
108    };
109    with_uses.into()
110}

112/// Build the inner support of the copper list.
113fn gen_culist_support(
114    runtime_plan: &CuExecutionLoop,
115    taskid_call_order: &[usize],
116    all_tasks_as_struct_member_name: &Vec<String>,
117) -> proc_macro2::TokenStream {
118    #[cfg(feature = "macro_debug")]
119    eprintln!("[Extract msgs types]");
120    let all_msgs_types_in_culist_order = extract_msg_types(runtime_plan);
121
122    let culist_size = all_msgs_types_in_culist_order.len();
123    let task_indices: Vec<_> = taskid_call_order
124        .iter()
125        .map(|i| syn::Index::from(*i))
126        .collect();
127
128    #[cfg(feature = "macro_debug")]
129    eprintln!("[build the copperlist struct]");
130    let msgs_types_tuple: TypeTuple = build_culist_tuple(&all_msgs_types_in_culist_order);
131
132    #[cfg(feature = "macro_debug")]
133    eprintln!("[build the copperlist tuple bincode support]");
134    let msgs_types_tuple_encode = build_culist_tuple_encode(&all_msgs_types_in_culist_order);
135    let msgs_types_tuple_decode = build_culist_tuple_decode(&all_msgs_types_in_culist_order);
136
137    #[cfg(feature = "macro_debug")]
138    eprintln!("[build the copperlist tuple debug support]");
139    let msgs_types_tuple_debug = build_culist_tuple_debug(&all_msgs_types_in_culist_order);
140
141    let collect_metadata_function = quote! {
142        pub fn collect_metadata<'a>(culist: &'a CuList) -> [&'a CuMsgMetadata; #culist_size] {
143            [#( &culist.msgs.0.#task_indices.metadata, )*]
144        }
145    };
146
147    let methods = itertools::multizip((all_tasks_as_struct_member_name, taskid_call_order)).map(
148        |(name, output_position)| {
149            let fn_name = format_ident!("get_{}_output", name);
150            let payload_type = all_msgs_types_in_culist_order[*output_position].clone();
151            let index = syn::Index::from(*output_position);
152            quote! {
153                pub fn #fn_name(&self) -> &CuMsg<#payload_type> {
154                    &self.0.#index
155                }
156            }
157        },
158    );
159
160    // This generates a way to get the metadata of every single message of a culist at low cost
161    quote! {
162        #collect_metadata_function
163
164        pub struct CuMsgs(pub #msgs_types_tuple);
165
166        pub type CuList = CopperList<CuMsgs>;
167
168        impl CuMsgs {
169            #(#methods)*
170
171            fn get_tuple(&self) -> &#msgs_types_tuple {
172                &self.0
173            }
174
175            fn get_tuple_mut(&mut self) -> &mut #msgs_types_tuple {
176                &mut self.0
177            }
178        }
179
180        // Adds the bincode support for the copper list tuple
181        #msgs_types_tuple_encode
182        #msgs_types_tuple_decode
183
184        // Adds the debug support
185        #msgs_types_tuple_debug
186    }
187}

189fn gen_sim_support(runtime_plan: &CuExecutionLoop) -> proc_macro2::TokenStream {
190    #[cfg(feature = "macro_debug")]
191    eprintln!("[Sim: Build SimEnum]");
192    let plan_enum: Vec<proc_macro2::TokenStream> = runtime_plan
193        .steps
194        .iter()
195        .map(|unit| match unit {
196            CuExecutionUnit::Step(step) => {
197                let enum_entry_name = config_id_to_enum(step.node.get_id().as_str());
198                let enum_ident = Ident::new(&enum_entry_name, proc_macro2::Span::call_site());
199                let inputs: Vec<Type> = step
200                    .input_msg_indices_types
201                    .iter()
202                    .map(|(_, t)| parse_str::<Type>(format!("CuMsg<{t}>").as_str()).unwrap())
203                    .collect();
204                let output: Option<Type> = step
205                    .output_msg_index_type
206                    .as_ref()
207                    .map(|(_, t)| parse_str::<Type>(format!("CuMsg<{t}>").as_str()).unwrap());
208                let no_output = parse_str::<Type>("CuMsg<()>").unwrap();
209                let output = output.as_ref().unwrap_or(&no_output);
210                quote! {
211                    #enum_ident(cu29::simulation::CuTaskCallbackState<'cl, (#(&'cl #inputs),*), &'cl mut #output>)
212                }
213            }
214            CuExecutionUnit::Loop(_) => {
215                todo!("Needs to be implemented")
216            }
217        })
218        .collect();
219    quote! {
220        pub enum SimStep<'cl> {
221            #(#plan_enum),*
222        }
223    }
224}

226/// Adds #[copper_runtime(config = "path", sim_mode = false/true)] to your application struct to generate the runtime.
227/// if sim_mode is omitted, it is set to false.
228/// This will add a "runtime" field to your struct and implement the "new" and "run" methods.
229#[proc_macro_attribute]
230pub fn copper_runtime(args: TokenStream, input: TokenStream) -> TokenStream {
231    #[cfg(feature = "macro_debug")]
232    eprintln!("[entry]");
233    let mut application_struct = parse_macro_input!(input as ItemStruct);
234    let mut config_file: Option<LitStr> = None;
235    let mut sim_mode = false;
236
237    // Custom parser for the attribute arguments
238    let attribute_config_parser = parser(|meta| {
239        if meta.path.is_ident("config") {
240            config_file = Some(meta.value()?.parse()?);
241            Ok(())
242        } else if meta.path.is_ident("sim_mode") {
243            // Check if `sim_mode` has an explicit value (true/false)
244            if meta.input.peek(syn::Token![=]) {
245                meta.input.parse::<syn::Token![=]>()?;
246                let value: syn::LitBool = meta.input.parse()?;
247                sim_mode = value.value();
248                Ok(())
249            } else {
250                // If no value is provided, default to true
251                sim_mode = true;
252                Ok(())
253            }
254        } else {
255            Err(meta.error("unsupported property"))
256        }
257    });
258
259    #[cfg(feature = "macro_debug")]
260    eprintln!("[parse]");
261    // Parse the provided args with the custom parser
262    parse_macro_input!(args with attribute_config_parser);
263
264    // Check if the config file was provided
265    let config_file = match config_file {
266        Some(file) => file.value(),
267        None => {
268            return return_error(
269                "Expected config file attribute like #[CopperRuntime(config = \"path\")]"
270                    .to_string(),
271            )
272        }
273    };
274
275    if !std::path::Path::new(&config_full_path(&config_file)).exists() {
276        return return_error(format!(
277            "The configuration file `{config_file}` does not exist. Please provide a valid path."
278        ));
279    }
280
281    let copper_config = match read_config(&config_file) {
282        Ok(cuconfig) => cuconfig,
283        Err(e) => return return_error(e.to_string()),
284    };
285    let copper_config_content = match read_to_string(config_full_path(config_file.as_str())) {
286        Ok(ok) => ok,
287        Err(e) => return return_error(format!("Could not read the config file (should not happen because we just succeeded just before). {e}"))
288    };
289
290    let mission = "default"; // FIXME(gbin) generate all the missions from the config.
291    let mission_mod =
292        parse_str::<Ident>(mission).expect("Could not make an identifier of the mission name");
293
294    #[cfg(feature = "macro_debug")]
295    eprintln!("[runtime plan for mission {mission}]");
296    let runtime_plan: CuExecutionLoop = match compute_runtime_plan(&copper_config) {
297        Ok(plan) => plan,
298        Err(e) => return return_error(format!("Could not compute runtime plan: {e}")),
299    };
300    #[cfg(feature = "macro_debug")]
301    eprintln!("{runtime_plan:?}");
302
303    #[cfg(feature = "macro_debug")]
304    eprintln!("[extract tasks ids & types]");
305    let (all_tasks_ids, all_tasks_cutype, all_tasks_types_names, all_tasks_types) =
306        extract_tasks_types(&copper_config);
307
308    let all_sim_tasks_types: Vec<Type> = all_tasks_ids
309        .iter()
310        .zip(&all_tasks_cutype)
311        .zip(&all_tasks_types)
312        .map(|((task_id, cutype), stype)| match cutype {
313            CuTaskType::Source => {
314                let msg_type = copper_config
315                    .get_node_output_msg_type(task_id.as_str(), None) // FIXME(gbin): Multimission
316                    .unwrap_or_else(|| panic!("CuSrcTask {task_id} should have an outgoing connection with a valid output msg type"));
317                let sim_task_name = format!("cu29::simulation::CuSimSrcTask<{msg_type}>");
318                parse_str(sim_task_name.as_str()).unwrap_or_else(|_| panic!("Could not build the placeholder for simulation: {sim_task_name}"))
319            }
320            CuTaskType::Regular => stype.clone(),
321            CuTaskType::Sink => {
322                let msg_type = copper_config
323                    .get_node_input_msg_type(task_id.as_str(), None) // FIXME(gbin): Multimission
324                    .unwrap_or_else(|| panic!("CuSinkTask {task_id} should have an incoming connection with a valid input msg type"));
325                let sim_task_name = format!("cu29::simulation::CuSimSinkTask<{msg_type}>");
326                parse_str(sim_task_name.as_str()).unwrap_or_else(|_| panic!("Could not build the placeholder for simulation: {sim_task_name}"))
327            }
328        })
329        .collect();
330
331    #[cfg(feature = "macro_debug")]
332    eprintln!("[build task tuples]");
333    // Build the tuple of all those types
334    // note the extraneous, at the end is to make the tuple work even if this is only one element
335    let task_types_tuple: TypeTuple = parse_quote! {
336        (#(#all_tasks_types),*,)
337    };
338
339    let task_types_tuple_sim: TypeTuple = parse_quote! {
340        (#(#all_sim_tasks_types),*,)
341    };
342
343    #[cfg(feature = "macro_debug")]
344    eprintln!("[build monitor type]");
345    let monitor_type = if let Some(monitor_config) = copper_config.get_monitor_config() {
346        let monitor_type = parse_str::<Type>(monitor_config.get_type())
347            .expect("Could not transform the monitor type name into a Rust type.");
348        quote! { #monitor_type }
349    } else {
350        quote! { NoMonitor }
351    };
352
353    // FIXME(gbin): generate that per mission
354    #[cfg(feature = "macro_debug")]
355    eprintln!("[build runtime field]");
356    // add that to a new field
357    let runtime_field: Field = if sim_mode {
358        parse_quote! {
359            copper_runtime: cu29::curuntime::CuRuntime<#mission_mod::CuSimTasks, #mission_mod::CuMsgs, #monitor_type, #DEFAULT_CLNB>
360        }
361    } else {
362        parse_quote! {
363            copper_runtime: cu29::curuntime::CuRuntime<#mission_mod::CuTasks, #mission_mod::CuMsgs, #monitor_type, #DEFAULT_CLNB>
364        }
365    };
366
367    let name = &application_struct.ident;
368
369    #[cfg(feature = "macro_debug")]
370    eprintln!("[match struct anonymity]");
371    match &mut application_struct.fields {
372        Named(fields_named) => {
373            fields_named.named.push(runtime_field);
374        }
375        Unnamed(fields_unnamed) => {
376            fields_unnamed.unnamed.push(runtime_field);
377        }
378        Fields::Unit => {
379            panic!("This struct is a unit struct, it should have named or unnamed fields. use struct Something {{}} and not struct Something;")
380        }
381    };
382
383    #[cfg(feature = "macro_debug")]
384    eprintln!("[gen instances]");
385
386    let task_sim_instances_init_code = all_sim_tasks_types.iter().enumerate().map(|(index, ty)| {
387        let additional_error_info = format!(
388            "Failed to get create instance for {}, instance index {}.",
389            all_tasks_types_names[index], index
390        );
391
392        quote! {
393        <#ty>::new(all_instances_configs[#index]).map_err(|e| e.add_cause(#additional_error_info))?
394        }
395    }).collect::<Vec<_>>();
396
397    // Generate the code to create instances of the nodes
398    // It maps the types to their index
399    let (task_instances_init_code,
400        start_calls,
401        stop_calls,
402        preprocess_calls,
403        postprocess_calls): (Vec<_>, Vec<_>, Vec<_>, Vec<_>, Vec<_>) = itertools::multiunzip(all_tasks_types
404        .iter()
405        .enumerate()
406        .map(|(index, ty)| {
407            let task_index = int2sliceindex(index as u32);
408            let task_enum_name = config_id_to_enum(&all_tasks_ids[index]);
409            let enum_name = Ident::new(&task_enum_name, proc_macro2::Span::call_site());
410            let additional_error_info = format!(
411                "Failed to get create instance for {}, instance index {}.",
412                all_tasks_types_names[index], index
413            );
414            (
415                quote! {
416                    #ty::new(all_instances_configs[#index]).map_err(|e| e.add_cause(#additional_error_info))?
417                },
418                {
419                    let monitoring_action = quote! {
420                        let decision = self.copper_runtime.monitor.process_error(#index, CuTaskState::Start, &error);
421                        match decision {
422                            Decision::Abort => {
423                                debug!("Start: ABORT decision from monitoring. Task '{}' errored out \
424                                during start. Aborting all the other starts.", #mission_mod::TASKS_IDS[#index]);
425                                return Ok(());
426
427                            }
428                            Decision::Ignore => {
429                                debug!("Start: IGNORE decision from monitoring. Task '{}' errored out \
430                                during start. The runtime will continue.", #mission_mod::TASKS_IDS[#index]);
431                            }
432                            Decision::Shutdown => {
433                                debug!("Start: SHUTDOWN decision from monitoring. Task '{}' errored out \
434                                during start. The runtime cannot continue.", #mission_mod::TASKS_IDS[#index]);
435                                return Err(CuError::new_with_cause("Task errored out during start.", error));
436                            }
437                        }
438                    };
439
440                    let call_sim_callback = if sim_mode {
441                        quote! {
442                            // Ask the sim if this task should be executed or overridden by the sim.
443                            let ovr = sim_callback(SimStep::#enum_name(cu29::simulation::CuTaskCallbackState::Start));
444
445                            let doit = if let cu29::simulation::SimOverride::Errored(reason) = ovr  {
446                                let error: CuError = reason.into();
447                                #monitoring_action
448                                false
449                           }
450                           else {
451                                ovr == cu29::simulation::SimOverride::ExecuteByRuntime
452                           };
453                        }
454                    } else {
455                        quote! {
456                            let doit = true;  // in normal mode always execute the steps in the runtime.
457                        }
458                    };
459

462                        #call_sim_callback
463                        if doit {
464                            let task = &mut self.copper_runtime.tasks.#task_index;

466                                #monitoring_action
467                            }
468                        }
469                    }
470                },
471                {
472                    let monitoring_action = quote! {

474                                match decision {
475                                    Decision::Abort => {
476                                        debug!("Stop: ABORT decision from monitoring. Task '{}' errored out \

481                                    Decision::Ignore => {

484                                    }

486                                        debug!("Stop: SHUTDOWN decision from monitoring. Task '{}' errored out \
487                                    during stop. The runtime cannot continue.", #mission_mod::TASKS_IDS[#index]);
488                                        return Err(CuError::new_with_cause("Task errored out during stop.", error));
489                                    }
490                                }
491                        };
492                    let call_sim_callback = if sim_mode {
493                        quote! {
494                            // Ask the sim if this task should be executed or overridden by the sim.
495                            let ovr = sim_callback(SimStep::#enum_name(cu29::simulation::CuTaskCallbackState::Stop));
496
497                            let doit = if let cu29::simulation::SimOverride::Errored(reason) = ovr  {
498                                let error: CuError = reason.into();
499                                #monitoring_action
500                                false
501                           }
502                           else {
503                                ovr == cu29::simulation::SimOverride::ExecuteByRuntime
504                           };
505                        }
506                    } else {
507                        quote! {
508                            let doit = true;  // in normal mode always execute the steps in the runtime.
509                        }
510                    };
511                    quote! {
512                        #call_sim_callback
513                        if doit {
514                            let task = &mut self.copper_runtime.tasks.#task_index;
515                            if let Err(error) = task.stop(&self.copper_runtime.clock) {
516                                #monitoring_action
517                            }
518                        }
519                    }
520                },
521                {
522                    let monitoring_action = quote! {
523                        let decision = self.copper_runtime.monitor.process_error(#index, CuTaskState::Preprocess, &error);
524                        match decision {
525                            Decision::Abort => {
526                                debug!("Preprocess: ABORT decision from monitoring. Task '{}' errored out \
527                                during preprocess. Aborting all the other starts.", #mission_mod::TASKS_IDS[#index]);
528                                return Ok(());
529
530                            }
531                            Decision::Ignore => {
532                                debug!("Preprocess: IGNORE decision from monitoring. Task '{}' errored out \
533                                during preprocess. The runtime will continue.", #mission_mod::TASKS_IDS[#index]);
534                            }
535                            Decision::Shutdown => {
536                                debug!("Preprocess: SHUTDOWN decision from monitoring. Task '{}' errored out \
537                                during preprocess. The runtime cannot continue.", #mission_mod::TASKS_IDS[#index]);
538                                return Err(CuError::new_with_cause("Task errored out during preprocess.", error));
539                            }
540                        }
541                    };
542                    let call_sim_callback = if sim_mode {
543                        quote! {
544                            // Ask the sim if this task should be executed or overridden by the sim.
545                            let ovr = sim_callback(SimStep::#enum_name(cu29::simulation::CuTaskCallbackState::Preprocess));
546
547                            let doit = if let cu29::simulation::SimOverride::Errored(reason) = ovr  {
548                                let error: CuError = reason.into();
549                                #monitoring_action

553                            };
554                        }

557                            let doit = true;  // in normal mode always execute the steps in the runtime.
558                        }
559                    };
560                    quote! {
561                        #call_sim_callback
562                        if doit {
563                            let task = &mut self.copper_runtime.tasks.#task_index;
564                            if let Err(error) = task.preprocess(&self.copper_runtime.clock) {
565                                #monitoring_action
566                            }
567                        }
568                    }
569                },
570                {
571                    let monitoring_action = quote! {
572                        let decision = self.copper_runtime.monitor.process_error(#index, CuTaskState::Postprocess, &error);
573                        match decision {
574                            Decision::Abort => {
575                                debug!("Postprocess: ABORT decision from monitoring. Task '{}' errored out \
576                                during postprocess. Aborting all the other starts.", #mission_mod::TASKS_IDS[#index]);
577                                return Ok(());
578

580                            Decision::Ignore => {
581                                debug!("Postprocess: IGNORE decision from monitoring. Task '{}' errored out \
582                                during postprocess. The runtime will continue.", #mission_mod::TASKS_IDS[#index]);
583                            }
584                            Decision::Shutdown => {
585                                debug!("Postprocess: SHUTDOWN decision from monitoring. Task '{}' errored out \
586                                during postprocess. The runtime cannot continue.", #mission_mod::TASKS_IDS[#index]);
587                                return Err(CuError::new_with_cause("Task errored out during postprocess.", error));
588                            }
589                        }
590                    };
591                    let call_sim_callback = if sim_mode {
592                        quote! {
593                            // Ask the sim if this task should be executed or overridden by the sim.
594                            let ovr = sim_callback(SimStep::#enum_name(cu29::simulation::CuTaskCallbackState::Postprocess));

1335/// Extract all the tasks types in their index order and their ids.
1336fn extract_tasks_types(
1337    copper_config: &CuConfig,
1338) -> (Vec<String>, Vec<CuTaskType>, Vec<String>, Vec<Type>) {
1339    let all_id_nodes = copper_config.get_all_nodes(None); // FIXME(gbin): Multimission
1340
1341    // Get all the tasks Ids
1342    let all_tasks_ids: Vec<String> = all_id_nodes
1343        .iter()
1344        .map(|(_, node)| node.get_id().to_string())
1345        .collect();
1346
1347    let all_task_cutype: Vec<CuTaskType> = all_id_nodes
1348        .iter()
1349        .map(|(id, _)| {
1350            find_task_type_for_id(
1351                copper_config
1352                    .get_graph(None)
1353                    .expect("Only implemented for Simple"),
1354                *id,
1355            )
1356        }) // FIXME(gbin): Multimission
1357        .collect();
1358
1359    // Collect all the type names used by our configs.
1360    let all_types_names: Vec<String> = all_id_nodes
1361        .iter()
1362        .map(|(_, node)| node.get_type().to_string())
1363        .collect();
1364
1365    // Transform them as Rust types
1366    let all_types: Vec<Type> = all_types_names
1367        .iter()
1368        .map(|name| {
1369            parse_str(name)
1370                .unwrap_or_else(|_| panic!("Could not transform {name} into a Task Rust type."))
1371        })
1372        .collect();
1373    (all_tasks_ids, all_task_cutype, all_types_names, all_types)
1374}

1399/// Builds the tuple of the CuList as a tuple off all the messages types.
1400fn build_culist_tuple(all_msgs_types_in_culist_order: &[Type]) -> TypeTuple {
1401    if all_msgs_types_in_culist_order.is_empty() {

1406        }

1410/// This is the bincode encoding part of the CuMsgs
1411fn build_culist_tuple_encode(all_msgs_types_in_culist_order: &[Type]) -> ItemImpl {
1412    let indices: Vec<usize> = (0..all_msgs_types_in_culist_order.len()).collect();
1413
1414    // Generate the `self.#i.encode(encoder)?` for each tuple index, including `()` types
1415    let encode_fields: Vec<_> = indices
1416        .iter()
1417        .map(|i| {
1418            let idx = syn::Index::from(*i);
1419            quote! { self.0.#idx.encode(encoder)?; }
1420        })
1421        .collect();
1422
1423    parse_quote! {
1424        impl Encode for CuMsgs {
1425            fn encode<E: Encoder>(&self, encoder: &mut E) -> Result<(), EncodeError> {
1426                #(#encode_fields)*
1427                Ok(())
1428            }
1429        }
1430    }
1431}

1433/// This is the bincode decoding part of the CuMsgs
1434fn build_culist_tuple_decode(all_msgs_types_in_culist_order: &[Type]) -> ItemImpl {
1435    let indices: Vec<usize> = (0..all_msgs_types_in_culist_order.len()).collect();
1436
1437    // Generate the `CuMsg::<T>::decode(decoder)?` for each tuple index
1438    let decode_fields: Vec<_> = indices
1439        .iter()
1440        .map(|i| {
1441            let t = &all_msgs_types_in_culist_order[*i];
1442            quote! { CuMsg::<#t>::decode(decoder)? }
1443        })
1444        .collect();
1445
1446    parse_quote! {
1447        impl Decode<()> for CuMsgs {
1448            fn decode<D: Decoder<Context=()>>(decoder: &mut D) -> Result<Self, DecodeError> {
1449                Ok(CuMsgs ((
1450                    #(#decode_fields),*
1451                )))
1452            }
1453        }
1454    }
1455}

1457fn build_culist_tuple_debug(all_msgs_types_in_culist_order: &[Type]) -> ItemImpl {
1458    let indices: Vec<usize> = (0..all_msgs_types_in_culist_order.len()).collect();
1459
1460    let debug_fields: Vec<_> = indices
1461        .iter()

1471                f.debug_tuple("CuMsgs")
1472                    #(#debug_fields)*
1473                    .finish()
1474            }
1475        }
1476    }
