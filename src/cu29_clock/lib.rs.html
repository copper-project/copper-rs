53        }

57    #[cfg(target_arch = "arm")]
58    pub fn read_raw_counter() -> u64 {
59        let mut counter_lo: u32;
60        let mut counter_hi: u32;
61        unsafe {

64                out(reg) counter_lo,

67        }

87    pub fn read_raw_counter() -> u64 {
88        // Fallback implementation for unsupported architectures
89        #[cfg(feature = "std")]
90        {
91            std::time::SystemTime::now()
92                .duration_since(std::time::UNIX_EPOCH)
93                .unwrap()
94                .as_nanos() as u64
95        }
96        #[cfg(not(feature = "std"))]

98            // For no-std environments on unsupported platforms, we need a compile-time error
99            compile_error!("Unsupported target architecture for high-precision timing");
100        }

103    // Frequency estimation for converting raw counter values to nanoseconds
104    static FREQUENCY_NS: AtomicU64 = AtomicU64::new(0);
105    static INIT_COUNTER: AtomicU64 = AtomicU64::new(0);
106    static INIT_TIME_NS: AtomicU64 = AtomicU64::new(0);
107
108    pub fn initialize_frequency() {
109        #[cfg(feature = "std")]
110        {
111            let start_counter = read_raw_counter();
112            let start_time = std::time::SystemTime::now()
113                .duration_since(std::time::UNIX_EPOCH)
114                .unwrap()
115                .as_nanos() as u64;
116
117            std::thread::sleep(std::time::Duration::from_millis(10));
118
119            let end_counter = read_raw_counter();
120            let end_time = std::time::SystemTime::now()
121                .duration_since(std::time::UNIX_EPOCH)
122                .unwrap()
123                .as_nanos() as u64;
124
125            let counter_diff = end_counter.saturating_sub(start_counter);
126            let time_diff_ns = end_time.saturating_sub(start_time);
127
128            if counter_diff > 0 {
129                let freq_ns =
130                    ((counter_diff as u128 * 1_000_000_000) / time_diff_ns as u128) as u64;
131                FREQUENCY_NS.store(freq_ns, Ordering::Relaxed);
132                INIT_COUNTER.store(start_counter, Ordering::Relaxed);
133                INIT_TIME_NS.store(start_time, Ordering::Relaxed);
134            }
135        }
136        #[cfg(not(feature = "std"))]
137        {
138            // In no-std environments, we can't do runtime frequency calibration
139            // Use a reasonable default frequency (e.g., 1GHz = 1 tick per nanosecond)
140            // This is a simplification and may not be accurate for all platforms
141            FREQUENCY_NS.store(1_000_000_000, Ordering::Relaxed);
142            INIT_COUNTER.store(0, Ordering::Relaxed);
143            INIT_TIME_NS.store(0, Ordering::Relaxed);
144        }
145    }
146
147    pub fn counter_to_nanos(counter: u64) -> u64 {
148        let freq = FREQUENCY_NS.load(Ordering::Relaxed);
149        if freq == 0 {
150            initialize_frequency();
151            let freq = FREQUENCY_NS.load(Ordering::Relaxed);
152            if freq == 0 {
153                return counter; // Fallback if frequency estimation fails
154            }
155        }
156
157        let init_counter = INIT_COUNTER.load(Ordering::Relaxed);
158        let init_time_ns = INIT_TIME_NS.load(Ordering::Relaxed);
159        let counter_diff = counter.saturating_sub(init_counter);
160
161        init_time_ns.saturating_add(((counter_diff as u128 * 1_000_000_000) / freq as u128) as u64)
162    }
163}

597/// Internal clock implementation that provides high-precision timing
598#[derive(Clone, Debug)]
599struct InternalClock {
600    // For real clocks, this stores the initialization time
601    // For mock clocks, this references the mock state
602    mock_state: Option<Arc<AtomicU64>>,
603}
604
605impl InternalClock {
606    fn new() -> Self {
607        // Initialize the frequency calibration
608        platform::initialize_frequency();
609        InternalClock { mock_state: None }
610    }
611
612    fn mock() -> (Self, Arc<AtomicU64>) {
613        let mock_state = Arc::new(AtomicU64::new(0));
614        let clock = InternalClock {
615            mock_state: Some(Arc::clone(&mock_state)),
616        };
617        (clock, mock_state)
618    }
619
620    fn now(&self) -> Instant {
621        if let Some(ref mock_state) = self.mock_state {
622            Instant(mock_state.load(Ordering::Relaxed))
623        } else {
624            Instant::now()
625        }
626    }
627
628    fn recent(&self) -> Instant {
629        // For simplicity, we use the same implementation as now()
630        // In a more sophisticated implementation, this could use a cached value
631        self.now()
632    }
633}

635/// A running Robot clock.
636/// The clock is a monotonic clock that starts at an arbitrary reference time.
637/// It is clone resilient, ie a clone will be the same clock, even when mocked.
638#[derive(Clone, Debug)]
639pub struct RobotClock {

641    ref_time: Instant,

644/// A mock clock that can be controlled by the user.
645#[derive(Debug, Clone)]
646pub struct RobotClockMock(Arc<AtomicU64>);
647
648impl RobotClockMock {
649    pub fn increment(&self, amount: Duration) {
650        let Self(mock_state) = self;
651        mock_state.fetch_add(amount.as_nanos() as u64, Ordering::Relaxed);
652    }
653
654    /// Decrements the time by the given amount.
655    /// Be careful this breaks the monotonicity of the clock.
656    pub fn decrement(&self, amount: Duration) {
657        let Self(mock_state) = self;
658        mock_state.fetch_sub(amount.as_nanos() as u64, Ordering::Relaxed);
659    }
660
661    /// Gets the current value of time.
662    pub fn value(&self) -> u64 {
663        let Self(mock_state) = self;
664        mock_state.load(Ordering::Relaxed)
665    }
666
667    /// A convenient way to get the current time from the mocking side.
668    pub fn now(&self) -> CuTime {
669        let Self(mock_state) = self;
670        CuDuration(mock_state.load(Ordering::Relaxed))
671    }
672
673    /// Sets the absolute value of the time.
674    pub fn set_value(&self, value: u64) {
675        let Self(mock_state) = self;
676        mock_state.store(value, Ordering::Relaxed);
677    }
678}

680impl RobotClock {
681    /// Creates a RobotClock using now as its reference time.
682    /// It will start at 0ns incrementing monotonically.
683    pub fn new() -> Self {
684        let clock = InternalClock::new();
685        let ref_time = clock.now();
686        RobotClock {
687            inner: clock,
688            ref_time,
689        }
690    }
691
692    /// Builds a monotonic clock starting at the given reference time.
693    pub fn from_ref_time(ref_time_ns: u64) -> Self {
694        let clock = InternalClock::new();
695        let ref_time = clock.now() - Duration::from_nanos(ref_time_ns);
696        RobotClock {
697            inner: clock,
698            ref_time,
699        }
700    }
701
702    /// Build a fake clock with a reference time of 0.
703    /// The RobotMock interface enables you to control all the clones of the clock given.
704    pub fn mock() -> (Self, RobotClockMock) {
705        let (clock, mock_state) = InternalClock::mock();
706        let ref_time = clock.now();
707        (
708            RobotClock {
709                inner: clock,
710                ref_time,
711            },
712            RobotClockMock(mock_state),
713        )
714    }
715
716    /// Now returns the time that passed since the reference time, usually the start time.
717    /// It is a monotonically increasing value.
718    #[inline]
719    pub fn now(&self) -> CuTime {
720        (self.inner.now() - self.ref_time).into()
721    }
722
723    /// A less precise but quicker time
724    #[inline]
725    pub fn recent(&self) -> CuTime {
726        (self.inner.recent() - self.ref_time).into()
727    }

937    #[test]
938    fn test_robot_clock_precision() {
939        // Test that RobotClock::now() and RobotClock::recent() return different values
940        // and that recent() is always <= now()
941        let clock = RobotClock::new();
942
943        // We can't guarantee the exact values, but we can check relationships
944        let recent = clock.recent();
945        let now = clock.now();
946
947        // recent() should be less than or equal to now()
948        assert!(recent <= now);
949
950        // Test precision of from_ref_time
951        let ref_time_ns = 1_000_000_000; // 1 second
952        let clock = RobotClock::from_ref_time(ref_time_ns);
953
954        // Clock should start at approximately ref_time_ns
955        let now = clock.now();
956        let now_ns: u64 = now.into();
957
958        // Allow reasonable tolerance for clock initialization time
959        let tolerance_ns = 50_000_000; // 50ms tolerance
960        assert!(now_ns >= ref_time_ns);
961        assert!(now_ns < ref_time_ns + tolerance_ns);
962    }
