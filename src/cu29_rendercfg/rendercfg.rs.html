1mod config;
2use clap::Parser;
3use config::read_configuration;
4pub use cu29_traits::*;
5use std::io::Cursor;
6use std::io::Write;
7use std::path::PathBuf;
8use std::process::{Command, Stdio};
9use tempfile::Builder;
10
11#[derive(Parser)]
12#[clap(author, version, about, long_about = None)]
13struct Args {
14    /// Config file name
15    #[clap(value_parser)]
16    config: PathBuf,
17    /// Open the SVG in the default system viewer
18    #[clap(long)]
19    open: bool,
20}
21
22/// Render the configuration file to a dot file then convert it to an SVG and optionally opens it with inkscape.
23fn main() -> std::io::Result<()> {
24    // Parse command line arguments
25    let args = Args::parse();
26
27    let config = read_configuration(args.config.to_str().unwrap())
28        .expect("Failed to read configuration file");
29    let mut content = Vec::<u8>::new();
30    {
31        let mut cursor = Cursor::new(&mut content);
32        config.render(&mut cursor);
33    }
34
35    // Generate SVG from DOT
36    let mut child = Command::new("dot")
37        .arg("-Tsvg")
38        .stdin(Stdio::piped())
39        .stdout(Stdio::piped())
40        .spawn()
41        .expect("Failed to start dot process");
42    child.wait().expect("Failed to wait for dot process");
43
44    {
45        let stdin = child.stdin.as_mut().expect("Failed to open stdin");
46        stdin.write_all(&content)?;
47    }
48
49    let output = child.wait_with_output().expect("Failed to read stdout");
50
51    if !output.status.success() {
52        std::process::exit(1);
53    }
54
55    let graph_svg = output.stdout;
56    if args.open {
57        // Create a temporary file to store the SVG
58        let mut temp_file = Builder::new().suffix(".svg").tempfile()?;
59        temp_file.write_all(graph_svg.as_slice())?;
60
61        // Open the SVG in the default system viewer
62        Command::new("inkscape") // xdg-open fails silently (while it works from a standard bash on the same file :shrug:)
63            .arg(temp_file.path())
64            .status()
65            .expect("failed to open SVG file");
66    } else {
67        // Write the SVG content to a file
68        let mut svg_file = std::fs::File::create("output.svg")?;
69        svg_file.write_all(graph_svg.as_slice())?;
70    }
71    Ok(())
72}
