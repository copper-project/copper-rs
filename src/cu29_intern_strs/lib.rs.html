16fn parent_n_times(path: &Path, n: usize) -> Option<PathBuf> {
17    let mut result = Some(path.to_path_buf());
18    for _ in 0..n {
19        result = result?.parent().map(PathBuf::from);
20    }
21    result
22}

24/// Convenience function to returns the default path for the log index directory.
25pub fn default_log_index_dir() -> PathBuf {
26    let outdir = std::env::var("LOG_INDEX_DIR").expect("no LOG_INDEX_DIR system variable set, be sure build.rs sets it, see cu29_log/build.rs for example.");
27    let outdir_path = Path::new(&outdir);
28    parent_n_times(outdir_path, 3).unwrap().join(INDEX_DIR_NAME)

33fn read_u32_le(bytes: &[u8]) -> u32 {
34    let b = <[u8; 4]>::try_from(bytes).unwrap();
35    u32::from_le_bytes(b)

38/// Reads all interned strings from the index at the specified path.
39/// The index is created at compile time within your project output directory.
40pub fn read_interned_strings(index: &Path) -> Result<Vec<String>> {
41    let mut all_strings = Vec::<String>::new();
42    let env =
43        Rkv::new::<Lmdb>(index).context("Could not open the string index. Check the path.")?;
44
45    let index_to_string = env
46        .open_single("index_to_string", StoreOptions::default())
47        .context("Could not open the index_to_string store")?;
48    let db_reader = env.read().unwrap();
49    let ri = index_to_string.iter_start(&db_reader);
50    let mut i = ri.expect("Failed to start iterator");
51    while let Some(Ok(v)) = i.next() {
52        let (k, v) = v;
53        let index = read_u32_le(k) as usize;
54
55        if let rkv::Value::Str(s) = v {
56            if all_strings.len() <= index {
57                all_strings.resize(index + 1, String::new());
58            }
59
60            all_strings[index] = s.to_string();
61        }
62    }
63    Ok(all_strings)
64}

79fn rkv() -> &'static Mutex<Rkv<LmdbEnvironment>> {
80    RKV.get_or_init(|| {
81        let target_dir = default_log_index_dir();
82
83        // Should never happen I believe.
84        if !target_dir.exists() {
85            fs::create_dir_all(&target_dir).unwrap();
86        }
87
88        #[cfg(feature = "macro_debug")]
89        {
90            build_log!(
91                "=================================================================================="
92            );
93            build_log!("Interned strings are stored in: {:?}", target_dir);
94            build_log!("   [r] is reused index and [n] is new index.");
95            build_log!(
96                "=================================================================================="
97            );
98        }
99
100        let env = Rkv::new::<Lmdb>(&target_dir).unwrap();
101        Mutex::new(env)
102    })
103}

105fn dbs() -> &'static Mutex<(SStore, SStore, SStore, MStore)> {
106    DBS.get_or_init(|| {
107        let env = rkv().lock().unwrap();
108
109        let counter = env.open_single("counter", StoreOptions::create()).unwrap();
110        let index_to_string = env
111            .open_single("index_to_string", StoreOptions::create())
112            .unwrap();
113        let string_to_index = env
114            .open_single("string_to_index", StoreOptions::create())
115            .unwrap();
116        let index_to_callsites = env
117            .open_multi("index_to_callsites", StoreOptions::create())
118            .unwrap();
119
120        Mutex::new((
121            counter,
122            index_to_string,
123            string_to_index,
124            index_to_callsites,
125        ))
126    })
127}

129pub fn intern_string(s: &str) -> Option<IndexType> {
130    let (counter_store, index_to_string, string_to_index, _) = &mut *dbs().lock().unwrap();
131    let index = {
132        let env = rkv().lock().unwrap();
133        // If this string already exists in the store, return the index
134        {
135            let reader = env.read().unwrap();
136            // check if log_string is already in the string_to_index store
137            if let Ok(Some(Value::U64(index))) = string_to_index.get(&reader, s) {
138                #[cfg(feature = "macro_debug")]
139                {
140                    build_log!("#{:0>3} [r] -> {}.", index, s);
141                }
142                return Some(index as IndexType);
143            };
144        }
145        let mut writer = env.write().unwrap();
146        let next_index = get_next_index(&mut writer, counter_store).unwrap();
147        // Insert the new string into the store
148        index_to_string
149            .put(&mut writer, next_index.to_le_bytes(), &Value::Str(s))
150            .unwrap();
151        string_to_index
152            .put(&mut writer, s, &Value::U64(next_index as u64))
153            .unwrap();
154        writer.commit().unwrap();
155        Some(next_index)
156    };
157    #[cfg(feature = "macro_debug")]
158    {
159        build_log!("#{:0>3} [n] -> {}.", index.unwrap(), s);
160    }
161    index
162}

169const COUNTER_KEY: &str = "__counter__";
170fn get_next_index(
171    writer: &mut Writer<LmdbRwTransaction>,
172    counter_store: &SStore,
173) -> Result<IndexType, Box<dyn std::error::Error>> {
174    let current_counter = match counter_store.get(writer, COUNTER_KEY)? {
175        Some(Value::U64(value)) => value as IndexType,
176        _ => 0,
177    };
178
179    let next_counter = current_counter + 1;
180    counter_store.put(writer, COUNTER_KEY, &Value::U64(next_counter as u64))?;
181    Ok(next_counter)
182}
