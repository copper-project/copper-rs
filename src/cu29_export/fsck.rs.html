1use crate::{copperlists_reader, keyframes_reader, structlog_reader};
2use cu29::prelude::UnifiedLoggerRead;
3use cu29::prelude::*;
4use cu29::{CopperListTuple, CuResult};
5use num_format::{Locale, ToFormattedString};
6use std::io::Cursor;
7
8pub(crate) fn check<P>(dl: &mut UnifiedLoggerRead, verbose: u8) -> Option<CuResult<()>>
9where
10    P: CopperListTuple,
11{
12    let header = dl.raw_main_header();
13
14    if verbose > 0 {
15        println!("Main header: \n{header}");
16    }
17    let mut overall_first_ts: OptionCuTime = OptionCuTime::none();
18    let mut last_ts: OptionCuTime = OptionCuTime::none();
19    let mut last_cl = 0;
20    let mut keyframes = 0;
21    let mut useful_size: usize = 0;
22    let mut structured_log_size: usize = 0;
23    let mut cls_size: usize = 0;
24    let mut kfs_size: usize = 0;
25    let mut sl_entries: usize = 0;
26
27    let result = loop {
28        // for _ in 0..4 {
29        let section = dl.raw_read_section();
30        match section {
31            Ok((header, content)) => {
32                useful_size += content.len();
33
34                if verbose > 0 {
35                    println!("Section: \n{header}");
36                }
37
38                match header.entry_type {
39                    UnifiedLogType::StructuredLogLine => {
40                        structured_log_size += content.len();
41                        let mut reader: Cursor<Vec<u8>> = Cursor::new(content);
42                        let iter = structlog_reader(&mut reader);
43                        for entry in iter {
44                            sl_entries += 1;
45                            if entry.is_err() {
46                                println!("Struct log #{sl_entries} is corrupted: {entry:?}");
47                            }
48                        }
49                    }
50                    UnifiedLogType::CopperList => {
51                        cls_size += content.len();
52
53                        let mut reader: Cursor<Vec<u8>> = Cursor::new(content);
54                        let iter = copperlists_reader::<P>(&mut reader);
55                        let mut first_cl = 0;
56                        let mut first_ts: OptionCuTime = OptionCuTime::none();
57                        for entry in iter {
58                            last_cl = entry.id;
59                            if first_ts.is_none() {
60                                first_cl = entry.id;
61                                first_ts = entry
62                                    .cumsgs()
63                                    .first()
64                                    .expect("Empty copperlist")
65                                    .metadata()
66                                    .process_time()
67                                    .start;
68                                if overall_first_ts.is_none() {
69                                    overall_first_ts = first_ts;
70                                }
71                            }
72                            let last_msg = *entry.cumsgs().last().expect("Empty copperlist");
73                            last_ts = last_msg.metadata().process_time().end;
74                        }
75                        if verbose > 0 {
76                            println!(
77                                "    CopperLists => OK (id range: [{first_cl}-{last_cl}] timerange: [{first_ts}-{last_ts}])"
78                            );
79                        }
80                    }
81                    UnifiedLogType::FrozenTasks => {
82                        kfs_size += content.len();
83                        let mut reader: Cursor<Vec<u8>> = Cursor::new(content);
84                        let iter = keyframes_reader(&mut reader);
85                        for entry in iter {
86                            keyframes += 1;
87                            if verbose > 0 {
88                                println!(
89                                    "    Keyframe CL/ts: {}/{} ",
90                                    entry.culistid, entry.timestamp
91                                );
92                            }
93                        }
94                    }
95                    UnifiedLogType::LastEntry => {
96                        if verbose > 0 {
97                            println!("Last Entry / EOF.");
98                            println!();
99                        }
100                        break Ok(());
101                    }
102                    UnifiedLogType::Empty => {
103                        println!("Error: Found an empty / Uninitialized section");
104                    }
105                }
106            }
107            Err(e) => {
108                println!("Failed to read section: {e}");
109                break Err(e);
110            }
111        }
112    };
113
114    let total_time = last_ts.unwrap() - overall_first_ts.unwrap();
115    let cl_rate = if last_cl != 0 {
116        let cl_time = total_time / last_cl;
117        1_000_000_000f64 / (cl_time.as_nanos() as f64)
118    } else {
119        0.0
120    };
121
122    let kf_rate = if keyframes != 0 {
123        let kf_time = total_time / keyframes as u64;
124        1_000_000_000f64 / (kf_time.as_nanos() as f64)
125    } else {
126        0.0
127    };
128
129    if result.is_ok() {
130        println!("The log checked out OK.");
131    } else {
132        println!("** The log is corrupted.");
133    }
134
135    let bytes_per_sec = useful_size as f64 * 1e9 / total_time.as_nanos() as f64;
136    let mib_per_sec = bytes_per_sec / (1024.0 * 1024.0);
137    let l = &Locale::en;
138    println!("        === Statistics ===");
139    println!("  Total time       -> {total_time}");
140    println!(
141        "  Total used size  -> {} bytes",
142        useful_size.to_formatted_string(l)
143    );
144    println!("  Logging rate     -> {mib_per_sec:.02} MiB/s (effective)");
145
146    println!();
147    println!("  # of CL          -> {}", last_cl.to_formatted_string(l));
148    println!(
149        "  CL rate          -> {}.{:02} Hz",
150        (cl_rate.trunc() as u64).to_formatted_string(&Locale::en),
151        (cl_rate.fract() * 100.0).round() as u64
152    );
153    println!(
154        "  CL total size    -> {} bytes",
155        cls_size.to_formatted_string(l)
156    );
157    println!();
158    println!("  # of Keyframes   -> {}", keyframes.to_formatted_string(l));
159    println!("  KF rate          -> {kf_rate:.2} Hz");
160    println!(
161        "  KF total size    -> {} bytes",
162        kfs_size.to_formatted_string(l)
163    );
164    println!();
165    println!(
166        "  # of SL entries  -> {}",
167        sl_entries.to_formatted_string(l)
168    );
169    println!(
170        "  SL total size    -> {} bytes",
171        structured_log_size.to_formatted_string(l)
172    );
173
174    None
175}
