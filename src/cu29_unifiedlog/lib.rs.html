61/// The main file header of the datalogger.
62#[derive(Encode, Decode, Debug)]
63pub struct MainHeader {
64    pub magic: [u8; 4],            // Magic number to identify the file.
65    pub first_section_offset: u16, // This is to align with a page at write time.
66    pub page_size: u16,
67}
68
69impl Display for MainHeader {
70    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult {
71        writeln!(
72            f,
73            "  Magic -> {:2x}{:2x}{:2x}{:2x}",
74            self.magic[0], self.magic[1], self.magic[2], self.magic[3]
75        )?;
76        writeln!(f, "  first_section_offset -> {}", self.first_section_offset)?;
77        writeln!(f, "  page_size -> {}", self.page_size)
78    }

81/// Each concurrent sublogger is tracked through a section header.
82/// They form a linked list of sections.
83/// The entry type is used to identify the type of data in the section.
84#[derive(Encode, Decode, Debug)]
85pub struct SectionHeader {
86    pub magic: [u8; 2], // Magic number to identify the section.
87    pub entry_type: UnifiedLogType,
88    pub section_size: u32, // offset from the first byte of this header to the first byte of the next header (MAGIC to MAGIC).
89    pub filled_size: u32,  // how much of the section is filled.
90}
91
92const MAX_HEADER_SIZE: usize = mem::size_of::<SectionHeader>() + 3usize; // 3 == additional worse case scenario for the 3 int variable encoding
93
94impl Display for SectionHeader {
95    fn fmt(&self, f: &mut Formatter<'_>) -> FmtResult {
96        writeln!(f, "    Magic -> {:2x}{:2x}", self.magic[0], self.magic[1])?;
97        writeln!(f, "    type -> {:?}", self.entry_type)?;
98        write!(
99            f,
100            "    use  -> {} / {}",
101            self.filled_size, self.section_size
102        )
103    }
104}
105
106impl Default for SectionHeader {
107    fn default() -> Self {
108        Self {
109            magic: SECTION_MAGIC,
110            entry_type: UnifiedLogType::Empty,
111            section_size: 0,
112            filled_size: 0,
113        }
114    }

122/// A SectionHandle is a handle to a section in the datalogger.
123/// It allows to track the lifecycle of a section of the datalogger.
124#[derive(Default)]
125pub struct SectionHandle {
126    section_header: SectionHeader,
127    buffer: &'static mut [u8], // This includes the encoded header for end of section patching.
128    used: u32,                 // this is the size of the used part of the buffer.
129}
130
131// This is for a placeholder to unsure an orderly cleanup as we dodge the borrow checker.
132
133impl SectionHandle {
134    // The buffer is considered static as it is a dedicated piece for the section.
135    pub fn create(section_header: SectionHeader, buffer: &'static mut [u8]) -> Self {
136        // here we assume with are passed a valid section.
137        if buffer[0] != SECTION_MAGIC[0] || buffer[1] != SECTION_MAGIC[1] {
138            panic!("Invalid section buffer, magic number not found");
139        }
140
141        if buffer.len() < MAX_HEADER_SIZE {
142            panic!(
143                "Invalid section buffer, too small: {}, it needs to be > {}",
144                buffer.len(),
145                MAX_HEADER_SIZE
146            );
147        }
148
149        Self {
150            section_header,
151            buffer,
152            used: 0,
153        }
154    }
155    pub fn get_user_buffer(&mut self) -> &mut [u8] {
156        &mut self.buffer[MAX_HEADER_SIZE + self.used as usize..]
157    }
158
159    pub fn update_header(&mut self) {
160        // no need to do anything if we never used the section.
161        if self.section_header.entry_type == UnifiedLogType::Empty || self.used == 0 {
162            return;
163        }
164        self.section_header.filled_size = self.used;
165    }
166}

175/// The writing interface to the unified logger.
176/// Writing is "almost" linear as various streams can allocate sections and track them until
177/// they drop them.
178pub trait UnifiedLogWrite: Send + Sync {
179    /// A section is a contiguous chunk of memory that can be used to write data.
180    /// It can store various types of data as specified by the entry_type.
181    /// The requested_section_size is the size of the section to allocate.
182    /// It returns a handle to the section that can be used to write data until
183    /// it is flushed with flush_section, it is then considered unmutable.
184    fn add_section(
185        &mut self,
186        entry_type: UnifiedLogType,
187        requested_section_size: usize,
188    ) -> SectionHandle;
189
190    /// Flush the given section to the underlying storage.
191    fn flush_section(&mut self, section: &mut SectionHandle);
192
193    /// Returns the current status of the unified logger.
194    fn status(&self) -> UnifiedLogStatus;
195}

197/// Read back a unified log linearly.
198pub trait UnifiedLogRead {
199    /// Read through the unified logger until it reaches the UnifiedLogType given in datalogtype.
200    /// It will return the byte array of the section if found.
201    fn read_next_section_type(&mut self, datalogtype: UnifiedLogType) -> CuResult<Option<Vec<u8>>>;
202
203    /// Read through the next section entry regardless of its type.
204    /// It will return the header and the byte array of the section.
205    /// Note the last Entry should be of UnifiedLogType::LastEntry if the log is not corrupted.
206    fn raw_read_section(&mut self) -> CuResult<(SectionHeader, Vec<u8>)>;
207}

209/// Create a new stream to write to the unifiedlogger.
210pub fn stream_write<E: Encode>(
211    logger: Arc<Mutex<impl UnifiedLogWrite>>,

218/// A wrapper around the unifiedlogger that implements the Write trait.
219struct LogStream<L: UnifiedLogWrite> {
220    entry_type: UnifiedLogType,
221    parent_logger: Arc<Mutex<L>>,
222    current_section: SectionHandle,
223    current_position: usize,
224    minimum_allocation_amount: usize,
225}
226
227impl<L: UnifiedLogWrite> LogStream<L> {
228    fn new(
229        entry_type: UnifiedLogType,
230        parent_logger: Arc<Mutex<L>>,
231        minimum_allocation_amount: usize,
232    ) -> Self {
233        #[cfg(feature = "std")]
234        let section = parent_logger
235            .lock()
236            .expect("Could not lock a section at MmapStream creation")
237            .add_section(entry_type, minimum_allocation_amount);
238
239        #[cfg(not(feature = "std"))]
240        let section = parent_logger
241            .lock()
242            .add_section(entry_type, minimum_allocation_amount);
243
244        Self {
245            entry_type,
246            parent_logger,
247            current_section: section,
248            current_position: 0,
249            minimum_allocation_amount,
250        }
251    }
252}

260impl<E: Encode, L: UnifiedLogWrite> WriteStream<E> for LogStream<L> {
261    fn log(&mut self, obj: &E) -> CuResult<()> {
262        let dst = self.current_section.get_user_buffer();
263        let result = encode_into_slice(obj, dst, standard());
264        match result {
265            Ok(nb_bytes) => {
266                self.current_position += nb_bytes;
267                self.current_section.used += nb_bytes as u32;
268                Ok(())
269            }
270            Err(e) => match e {
271                EncodeError::UnexpectedEnd => {
272                    #[cfg(feature = "std")]
273                    let logger_guard = self.parent_logger.lock();
274
275                    #[cfg(not(feature = "std"))]
276                    let mut logger_guard = self.parent_logger.lock();
277
278                    #[cfg(feature = "std")]
279                    let mut logger_guard =
280                        match logger_guard {
281                            Ok(g) => g,
282                            Err(_) => return Err(
283                                "Logger mutex poisoned while reporting EncodeError::UnexpectedEnd"
284                                    .into(),
285                            ), // It will retry but at least not completely crash.
286                        };
287
288                    logger_guard.flush_section(&mut self.current_section);
289                    self.current_section =
290                        logger_guard.add_section(self.entry_type, self.minimum_allocation_amount);
291
292                    let result = encode_into_slice(
293                        obj,
294                        self.current_section.get_user_buffer(),
295                        standard(),
296                    )
297                    .expect(
298                        "Failed to encode object in a newly minted section. Unrecoverable failure.",
299                    ); // If we fail just after creating a section, there is not much we can do, we need to bail.
300                    self.current_position += result;
301                    self.current_section.used += result as u32;
302                    Ok(())
303                }
304                _ => {
305                    let err =
306                        <&str as Into<CuError>>::into("Unexpected error while encoding object.")
307                            .add_cause(e.to_string().as_str());
308                    Err(err)
309                }
310            },
311        }
312    }
313}

315impl<L: UnifiedLogWrite> Drop for LogStream<L> {
316    fn drop(&mut self) {
317        #[cfg(feature = "std")]
318        let logger_guard = self.parent_logger.lock();
319
320        #[cfg(not(feature = "std"))]
321        let mut logger_guard = self.parent_logger.lock();
322
323        #[cfg(feature = "std")]
324        let mut logger_guard = match logger_guard {
325            Ok(g) => g,
326            Err(_) => return,
327        };
328        logger_guard.flush_section(&mut self.current_section);
329        #[cfg(feature = "std")]
330        if !std::thread::panicking() {
331            eprintln!("⚠️ MmapStream::drop: logger mutex poisoned");
332        }
333    }
